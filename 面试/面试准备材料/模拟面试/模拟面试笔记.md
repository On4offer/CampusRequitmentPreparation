**.xls 和 .xlsx 解析的底层 API 与内存机制区别**

- 底层 API：`.xls`（Excel 97-2003）对应的是 POI 的 `HSSFWorkbook`；`.xlsx`（Excel 2007+）对应的是 `XSSFWorkbook`。`WorkbookFactory` 是一个**工厂工具类**，作用是根据输入流自动判断文件格式，创建对应的 `HSSFWorkbook` 或 `XSSFWorkbook` 实例，并非具体的解析实现类。
- 内存机制区别：`HSSFWorkbook` 解析二进制格式，**最大支持 65536 行、256 列**，且全程将数据加载到内存，内存占用高；`XSSFWorkbook` 解析 XML 压缩格式，无行数和列数上限，但同样是**一次性加载全量数据到内存**，处理大数据量时极易引发 OOM。

**10 万行数据 OOM 的根本原因与优化方案**

- 根本原因：`HSSFWorkbook`/`XSSFWorkbook` 会将 Excel 的所有行、单元格数据一次性加载到 JVM 堆内存中，10 万行数据的单元格对象会占用大量内存空间，超出 JVM 堆内存阈值，从而触发 OOM。
- 优化方案：
  - **方案一（基于 POI 原生）**：使用 `SXSSFWorkbook`（XSSF 的低内存版本），它通过**临时文件缓存**机制，只在内存中保留指定数量的行（默认 100 行，可通过`setRowAccessWindowSize()`调整），超出部分写入磁盘临时文件，大幅降低内存占用。
  - **方案二（主流开源工具）**：使用阿里的 `EasyExcel`，它采用**流式解析**，逐行读取并处理数据，不缓存全量数据，内存占用恒定，是处理大数据量 Excel 导入的首选方案。
  - **辅助优化**：分批次读取 + 批量入库（如 MyBatis-Plus 的`saveBatch`）、调整 JVM 堆内存参数（-Xmx4G 等）、关闭 POI 的单元格类型自动检测。

```java
// 写入本地文件方法
private void writeLogToFile(SysLoginLog log) throws IOException {
    String logDir = "/var/log/audit/";
    String fileName = "audit_log_" + LocalDate.now() + ".log";
    File file = new File(logDir + fileName);
    
    // 确保目录存在
    file.getParentFile().mkdirs();
    
    // 追加写入（线程安全）
    try (FileWriter writer = new FileWriter(file, true);
         BufferedWriter bw = new BufferedWriter(writer)) {
        String logLine = JSON.toJSONString(log) + "\n";
        synchronized (this) {  // 文件写入需要同步
            bw.write(logLine);
            bw.flush();
        }
    }
}
```

### 方案2：消息队列持久化（第二层防护，更可靠）

#### 2.1 实现思路

#### 2.2 具体实现（以RabbitMQ为例）

```java
@Autowired
private RabbitTemplate rabbitTemplate;

public void recordLog(SysLoginLog log) {
    try {
        // 发送到消息队列（持久化）
        rabbitTemplate.convertAndSend(
            "audit.log.exchange",  // 交换机
            "audit.log.routing.key",  // 路由键
            log,
            message -> {
                // 设置消息持久化
                message.getMessageProperties().setDeliveryMode(MessageDeliveryMode.PERSISTENT);
                return message;
            }
        );
    } catch (Exception e) {
        log.error("发送日志到消息队列失败，降级到本地文件", e);
        // 降级到本地文件
        writeLogToFile(log);
    }
}

// 消费者：从消息队列消费并保存到数据库
@RabbitListener(queues = "audit.log.queue")
public void consumeLog(SysLoginLog log) {
    try {
        loginLogService.save(log);
        // 手动确认消息（只有保存成功才确认）
    } catch (Exception e) {
        log.error("保存日志失败，消息会重新入队", e);
        // 不确认消息，消息会重新入队或进入死信队列
        throw e;  // 抛出异常，触发重试

}
```

```java
public void recordLogWithRetry(SysLoginLog log) {
    int maxRetries = 3;
    int retryCount = 0;
    
    while (retryCount < maxRetries) {
        try {
            loginLogService.save(log);
            return;  // 成功，退出
        } catch (Exception e) {
            retryCount++;
            if (retryCount >= maxRetries) {
                // 重试失败，降级到本地文件
                log.error("重试{}次后仍然失败，降级到本地文件", maxRetries, e);
                writeLogToFile(log);
            } else {
                // 等待后重试
                try {
                    Thread.sleep(1000 * retryCount);  // 指数退避
                } catch (InterruptedException ie) {
                    Thread.currentThread().interrupt();
                    writeLogToFile(log);
                    return;
                }
            }
        }
    }
}
```

```java
public void compensateMissingLogs() {

        // 1. 查询最近24小时的业务操作（有记录但可能没有日志）
        LocalDateTime startTime = LocalDateTime.now().minusDays(1);
        List<BusinessOperation> operations = businessService.getOperationsAfter(startTime);
```

```java
// 补偿表结构
CREATE TABLE log_compensation (
    id VARCHAR(36) PRIMARY KEY,
    operation_id VARCHAR(36) NOT NULL,
    log_data TEXT NOT NULL,
    retry_count INT DEFAULT 0,
    max_retries INT DEFAULT 5,
    status VARCHAR(20) DEFAULT 'PENDING',
    create_time DATETIME,
    update_time DATETIME
);

// 日志记录失败时，写入补偿表
public void recordLog(SysOperationLog log) {
    try {
        logService.save(log);
    } catch (Exception e) {
        log.error("日志记录失败，写入补偿表", e);
        // 写入补偿表
        LogCompensation compensation = new LogCompensation();
        compensation.setId(UUIDUtil.getUUID());
        compensation.setOperationId(log.getOperationId());
        compensation.setLogData(JSON.toJSONString(log));
        compensation.setStatus("PENDING");
        compensationService.save(compensation);
    }
}

// 定时任务：处理补偿表
@Scheduled(fixedDelay = 60000)  // 每分钟执行
public void processCompensation() {
    List<LogCompensation> compensations = compensationService.findPending();
    
    for (LogCompensation compensation : compensations) {
        try {
            // 解析日志数据
            SysOperationLog log = JSON.parseObject(
                compensation.getLogData(), 
                SysOperationLog.class
            );
            
            // 重试保存
            logService.save(log);
            
            // 标记为成功
            compensation.setStatus("SUCCESS");
            compensationService.update(compensation);
            
        } catch (Exception e) {
            // 增加重试次数
            compensation.setRetryCount(compensation.getRetryCount() + 1);
            
            if (compensation.getRetryCount() >= compensation.getMaxRetries()) {
                // 超过最大重试次数，标记为失败
                compensation.setStatus("FAILED");
                log.error("补偿失败，操作ID: {}", compensation.getOperationId(), e);
            }
            
            compensationService.update(compensation);
        }
    }
}
```

### 方案2：消息队列持久化（第二层，可靠传输）

#### 2.1 实现思路

```java
@Autowired
private RabbitTemplate rabbitTemplate;

@Transactional
public Result saveRepertoryEnter(RepertoryEnterDTO dto) {
    // 1. 保存到本地数据库
    Repertoryenters enter = repertoryenterService.save(dto);
    
    // 2. 发送到消息队列（持久化）
    try {
        rabbitTemplate.convertAndSend(
            "sync.repertoryenter.exchange",
            "sync.repertoryenter.routing.key",
            enter,
            message -> {
                // 设置消息持久化
                message.getMessageProperties().setDeliveryMode(
                    MessageDeliveryMode.PERSISTENT
                );
                // 设置消息ID（用于去重）
                message.getMessageProperties().setMessageId(enter.getId());
                return message;
            }
        );
        
        // 标记为同步中
        enter.setSyncStatus(1);
        repertoryenterService.update(enter);
        
    } catch (Exception e) {
        log.error("发送消息队列失败，降级到本地存储", e);
        // 降级：标记为未同步，等待定时任务同步
        enter.setSyncStatus(0);
        repertoryenterService.update(enter);
    }
    
    return Result.success();
}
```

```java
                Consumer.from("sync-group", "consumer-1"),
                StreamReadOptions.empty().count(10),
                StreamOffset.create("stream:sync:repertoryenter", 
                    ReadOffset.lastConsumed())

        for (MapRecord<String, Object, Object> record : records) {

                syncToHeadquarters(enter);

            }
        }
```

## 三、断网恢复后的同步机制

### 3.1 定时任务补偿
```java
@Scheduled(fixedDelay = 60000)  // 每分钟执行一次
public void syncPendingData() {
    try {
        // 1. 查询未同步的数据
        List<Repertoryenters> pendingList = repertoryenterService
            .findBySyncStatus(0);  // 未同步
        
        // 2. 检查网络连接
        if (!checkNetworkConnection()) {
            log.warn("网络未连接，跳过同步");
            return;
        }
        
        // 3. 批量同步
        for (Repertoryenters enter : pendingList) {
            try {
                // 标记为同步中
                enter.setSyncStatus(1);
                repertoryenterService.update(enter);
                
                // 同步到总部
                syncToHeadquarters(enter);
                
                // 同步成功
                enter.setSyncStatus(2);
                enter.setSyncTime(new Date());
                repertoryenterService.update(enter);
                
            } catch (Exception e) {
                // 同步失败
                enter.setSyncStatus(3);
                enter.setSyncRetryCount(enter.getSyncRetryCount() + 1);
                repertoryenterService.update(enter);
                
                log.error("同步失败，入库单ID: {}", enter.getId(), e);
            }
        }
        
        // 4. 处理本地文件中的数据（如果有）
        syncFromLocalFile();
        
    } catch (Exception e) {
        log.error("同步任务执行失败", e);
    }
}
```

### 3.2 重试机制
```java
// 同步时添加重试机制
private void syncToHeadquarters(Repertoryenters enter) {
    int maxRetries = 3;
    int retryCount = 0;
    
    while (retryCount < maxRetries) {
        try {
            // 调用总部接口
            String url = headquartersUrl + "/api/repertoryenter/sync";
            HttpResult result = httpClient.post(url, enter);
            
            if (result.isSuccess()) {
                return;  // 成功，退出
            }
            
        } catch (Exception e) {
            retryCount++;
            if (retryCount >= maxRetries) {
                throw e;  // 重试失败，抛出异常
            }
            // 等待后重试（指数退避）
            try {
                Thread.sleep(1000 * (int)Math.pow(2, retryCount));
            } catch (InterruptedException ie) {
                Thread.currentThread().interrupt();
                throw new RuntimeException(ie);
            }
        }
    }
}
```

## 四、推荐方案组合

### 4.1 完整方案
```
1. 门店保存数据 → 本地数据库（标记未同步）
2. 发送到消息队列（持久化）
3. 消息队列消费者 → 同步到总部
4. 如果消息队列失败 → 降级到本地文件
5. 定时任务 → 检查未同步数据并补偿
6. 断网恢复后 → 自动同步
```

### 4.2 方案选择
- 如果项目已有消息队列：优先使用消息队列 + 本地数据库
- 如果项目已有Redis：可以使用Redis Stream + 本地数据库
- 如果资源有限：使用本地数据库 + 定时任务 + 本地文件降级

```java
@Service
public class DataConsistencyChecker {
    
    @Autowired
    private RepertoryenterService localService;  // 门店服务
    
    @Autowired
    private HeadquartersApiClient headquartersClient;  // 总部API客户端
    
    /**
     * 检查数据一致性
     */
    public ConsistencyCheckResult checkConsistency(String clinicId, 
                                                    LocalDateTime startTime, 
                                                    LocalDateTime endTime) {
        ConsistencyCheckResult result = new ConsistencyCheckResult();
        
        // 1. 查询门店数据
        List<Repertoryenters> localData = localService
            .findByClinicIdAndTimeRange(clinicId, startTime, endTime);
        
        // 2. 查询总部数据
        List<RepertoryEnterDTO> headquartersData = headquartersClient
            .queryByClinicIdAndTimeRange(clinicId, startTime, endTime);
        
        // 3. 对比数据
        Set<String> localIds = localData.stream()
            .map(Repertoryenters::getId)
            .collect(Collectors.toSet());
        
        Set<String> headquartersIds = headquartersData.stream()
            .map(RepertoryEnterDTO::getId)
            .collect(Collectors.toSet());
        
        // 4. 找出缺失的数据
        // 门店有但总部没有的（需要补传）
        List<String> missingInHeadquarters = localData.stream()
            .filter(e -> !headquartersIds.contains(e.getId()))
            .map(Repertoryenters::getId)
            .collect(Collectors.toList());
        
        // 总部有但门店没有的（可能是删除，需要人工审核）
        List<String> missingInLocal = headquartersData.stream()
            .filter(dto -> !localIds.contains(dto.getId()))
            .map(RepertoryEnterDTO::getId)
            .collect(Collectors.toList());
        
        result.setMissingInHeadquarters(missingInHeadquarters);
        result.setMissingInLocal(missingInLocal);
        result.setTotalLocal(localData.size());
        result.setTotalHeadquarters(headquartersData.size());
        
        return result;
    }
}
```

## 一、缓存穿透解决方案

### 核心思路
查询不存在的数据时，缓存空值，避免重复查询数据库。

### 代码实现
```java
public <R, ID> R queryWithPassThrough(
        String keyPrefix, ID id, Class<R> type, 
        Function<ID, R> dbFallback, Long time, TimeUnit unit) {
    String key = keyPrefix + id;
    
    // 1. 从Redis查询缓存
    String json = stringRedisTemplate.opsForValue().get(key);
    
    // 2. 判断是否存在（关键：区分空值和不存在）
    if (StrUtil.isNotBlank(json)) {
        // ✅ 缓存命中，直接返回
        return JSONUtil.toBean(json, type);
    }
    
    // ✅ 关键逻辑1：判断是否是空值占位符
    if (json != null) {  // json == "" 表示空值占位符
        // ✅ 空值占位符，直接返回null，不再查询数据库
        return null;
    }
    
    // 3. 缓存不存在，查询数据库
    R r = dbFallback.apply(id);
    
    // 4. 数据库不存在（关键：缓存空值）
    if (r == null) {
        // ✅ 关键逻辑2：缓存空值，设置较短的TTL（2分钟）
        stringRedisTemplate.opsForValue().set(
            key, "", 
            CACHE_NULL_TTL,  // 2分钟
            TimeUnit.MINUTES
        );
        return null;
    }
    
    // 5. 数据库存在，写入缓存
    this.set(key, r, time, unit);
    return r;
}
```

- ✅ 关键逻辑1：区分空值（""）和不存在（null），空值直接返回
- ✅ 关键逻辑2：数据库不存在时缓存空值，避免重复查询
- ✅ 空值TTL较短（2分钟），避免占用过多内存

## 二、缓存击穿解决方案

### 方案1：互斥锁方案

### 核心思路
热点key过期时，使用分布式锁保证只有一个线程重建缓存。

### 代码实现
```java
public <R, ID> R queryWithMutex(
        String keyPrefix, ID id, Class<R> type,
        Function<ID, R> dbFallback, Long time, TimeUnit unit) {
    String key = keyPrefix + id;
    
    // 1. 从Redis查询缓存
    String json = stringRedisTemplate.opsForValue().get(key);
    if (StrUtil.isNotBlank(json)) {
        return JSONUtil.toBean(json, type);
    }
    
    // ✅ 关键逻辑1：判断空值占位符
    if (json != null) {
        return null;
    }
    
    // 2. 缓存不存在，尝试获取锁
    String lockKey = LOCK_SHOP_KEY + id;
    R r = null;
    
    try {
        // ✅ 关键逻辑2：获取分布式锁（SETNX，10秒过期）
        boolean isLock = tryLock(lockKey);
        
        if (!isLock) {
            // ✅ 关键逻辑3：获取锁失败，休眠后重试（避免大量请求同时查询数据库）
            Thread.sleep(50);
            return queryWithMutex(keyPrefix, id, type, dbFallback, time, unit);
        }
        
        // ✅ 关键逻辑4：获取锁成功，双重检查（防止其他线程已重建）
        json = stringRedisTemplate.opsForValue().get(key);
        if (StrUtil.isNotBlank(json)) {
            return JSONUtil.toBean(json, type);
        }
        
        // 3. 查询数据库
        r = dbFallback.apply(id);
        
        // 4. 处理空值
        if (r == null) {
            stringRedisTemplate.opsForValue().set(key, "", CACHE_NULL_TTL, TimeUnit.MINUTES);
            return null;
        }
        
        // 5. 重建缓存
        this.set(key, r, time, unit);
        
    } catch (InterruptedException e) {
        throw new RuntimeException(e);
    } finally {
        // ✅ 关键逻辑5：释放锁
        unlock(lockKey);
    }
    
    return r;
}

// 获取锁
private boolean tryLock(String key) {
    // ✅ SETNX + 过期时间，原子操作
    Boolean flag = stringRedisTemplate.opsForValue()
        .setIfAbsent(key, "1", 10, TimeUnit.SECONDS);
    return BooleanUtil.isTrue(flag);
}
```

### 关键逻辑标注
- ✅ 关键逻辑2：使用SETNX获取分布式锁，防止并发重建
- ✅ 关键逻辑3：获取锁失败时重试，避免大量请求同时查询数据库
- ✅ 关键逻辑4：双重检查，防止重复重建
- ✅ 关键逻辑5：finally中释放锁，确保锁被释放

### 方案2：逻辑过期方案

```java
public <R, ID> R queryWithLogicalExpire(
        String keyPrefix, ID id, Class<R> type,
        Function<ID, R> dbFallback, Long time, TimeUnit unit) {
    String key = keyPrefix + id;
    
    // 1. 从Redis查询缓存
    String json = stringRedisTemplate.opsForValue().get(key);
    
    // ✅ 关键逻辑1：缓存不存在，直接返回null（需要预热）
    if (StrUtil.isBlank(json)) {
        return null;
    }
    
    // 2. 反序列化，获取数据和逻辑过期时间
    RedisData redisData = JSONUtil.toBean(json, RedisData.class);
    R r = JSONUtil.toBean((JSONObject) redisData.getData(), type);
    LocalDateTime expireTime = redisData.getExpireTime();
    
    // ✅ 关键逻辑2：判断逻辑过期时间
    if (expireTime.isAfter(LocalDateTime.now())) {
        // ✅ 未过期，直接返回（即使数据可能已更新，但保证可用性）
        return r;
    }
    
    // ✅ 关键逻辑3：已过期，尝试获取锁重建
    String lockKey = LOCK_SHOP_KEY + id;
    boolean isLock = tryLock(lockKey);
    
    if (isLock) {
        // ✅ 关键逻辑4：获取锁成功，异步重建缓存（不阻塞主线程）
        CACHE_REBUILD_EXECUTOR.submit(() -> {
            try {
                // 查询数据库
                R newR = dbFallback.apply(id);
                // 重建缓存（设置新的逻辑过期时间）
                this.setWithLogicalExpire(key, newR, time, unit);
            } catch (Exception e) {
                log.error("重建缓存失败", e);
            } finally {
                unlock(lockKey);
            }
        });
    }
    
    // ✅ 关键逻辑5：返回过期数据（保证可用性，不阻塞用户请求）
    return r;
}

// 设置逻辑过期时间
public void setWithLogicalExpire(String key, Object value, Long time, TimeUnit unit) {
    RedisData redisData = new RedisData();
    redisData.setData(value);
    // ✅ 关键逻辑6：设置逻辑过期时间（当前时间 + 过期时长）
    redisData.setExpireTime(
        LocalDateTime.now().plusSeconds(unit.toSeconds(time))
    );
    // ✅ 不设置Redis过期时间，永久存储
    stringRedisTemplate.opsForValue().set(key, JSONUtil.toJsonStr(redisData));
}
```

## 一、TTL设计原则

### 1.1 基本原则
- 数据更新频率：更新频繁的数据，TTL应该短一些
- 数据重要性：重要的数据，TTL可以长一些（减少数据库压力）
- 缓存命中率：希望提高命中率，TTL可以长一些
- 数据一致性要求：一致性要求高，TTL应该短一些

### 1.2 权衡考虑
- TTL短：数据更新及时，但数据库压力大，缓存命中率低
- TTL长：数据库压力小，缓存命中率高，但数据可能不及时

```java
            randomRange = 30;  // ±15分钟
            break;
        case 1:  // 低热度
        default:
            baseTTL = 90;  // 90分钟
            randomRange = 60;  // ±30分钟
            break;
    }
    
    // 3. 添加随机值（防止缓存雪崩）
    Random random = new Random();
    long randomTTL = baseTTL + random.nextInt((int)randomRange) - randomRange / 2;
    
    // 4. 设置缓存
    stringRedisTemplate.opsForValue().set(
        key, 
        JSONUtil.toJsonStr(value), 
        randomTTL, 
        TimeUnit.MINUTES
    );
}
```

## 四、设计依据说明

### 4.1 高热度店铺TTL较短的原因
1. 访问频繁：
   - 日访问量 > 1000次
   - 即使TTL短（20-30分钟），缓存命中率仍然很高
   - 例如：25分钟TTL，每分钟访问40次，命中率 = 40/40 = 100%

2. 数据更新可能频繁：
   - 高热度店铺信息可能经常更新
   - 较短的TTL保证数据及时性

3. 数据库压力考虑：
   - 虽然TTL短，但因为访问频繁，大部分请求仍然命中缓存
   - 减少数据库压力

### 4.2 低热度店铺TTL较长的原因
1. 访问频率低：
   - 日访问量 < 100次
   - 即使TTL长（60-120分钟），也不会影响用户体验
   - 因为访问少，数据更新需求也少

2. 减少数据库压力：
   - 虽然访问少，但也要减少数据库查询
   - 长TTL提高缓存命中率

3. 资源利用：
   - 低热度店铺占用内存少，可以设置较长TTL
   - 提高整体缓存利用率

### 4.3 随机TTL的作用
1. 防止缓存雪崩：
   - 如果所有店铺都设置相同的TTL（如30分钟）
   - 预热时同时设置，会同时过期
   - 添加随机值后，过期时间分散，避免同时过期

2. 随机范围设计：
   - 高热度：±5分钟（20-30分钟）
   - 中热度：±15分钟（30-60分钟）
   - 低热度：±30分钟（60-120分钟）
   - 范围越大，分散效果越好，但TTL波动也越大

```java
private ThreadPoolExecutor excelImportExecutor;

public void importExcel(MultipartFile file) {
    List<Product> excelProducts = parseExcel(file);
    
    // 分批处理，每批1000条
    int batchSize = 1000;
    CountDownLatch latch = new CountDownLatch(
        (excelProducts.size() + batchSize - 1) / batchSize
    );
    
    for (int i = 0; i < excelProducts.size(); i += batchSize) {
        List<Product> batch = excelProducts.subList(
            i, Math.min(i + batchSize, excelProducts.size())
        );
        
        excelImportExecutor.submit(() -> {
            try {
                processBatch(batch); // 批量查询+匹配
            } finally {
                latch.countDown();
            }
        });
    }
    
    latch.await(30, TimeUnit.SECONDS);
}
```

### 题目3：登录拦截器实现

**原题：** 你是怎么实现拦截的？是用了 Spring MVC 的 Interceptor 还是 Filter？两者的区别是什么？为什么选这个？

**你的回答：** 拦截器，粒度更细，可以精确到方法层面，更灵活。filter一般用作全局日志相关的，不适合这里。

**评判：**
- 优点：正确选择了Interceptor
- 不足：未说明Filter和Interceptor的区别，理由不够充分

**标准答案：**

**Filter vs Interceptor：**

| 特性         | Filter（过滤器）                    | Interceptor（拦截器）                        |
| ------------ | ----------------------------------- | -------------------------------------------- |
| **所属框架** | Servlet规范                         | Spring MVC框架                               |
| **执行时机** | 在DispatcherServlet之前             | 在DispatcherServlet之后，Controller之前/之后 |
| **作用范围** | 所有请求（包括静态资源）            | 只拦截Controller方法                         |
| **获取信息** | 只能获取HttpServletRequest/Response | 可以获取HandlerMethod、ModelAndView等        |
| **异常处理** | 无法直接处理Controller异常          | 可以处理Controller异常                       |
| **依赖注入** | 需要特殊配置                        | 可以直接使用@Autowired                       |

**代码实现：**

```java
// Interceptor实现
@Component
public class LoginInterceptor implements HandlerInterceptor {
    
    @Override
    public boolean preHandle(HttpServletRequest request, 
                           HttpServletResponse response, 
                           Object handler) throws Exception {
        // 1. 判断是否需要拦截（静态资源不拦截）
        if (!(handler instanceof HandlerMethod)) {
            return true;
        }
        
        // 2. 检查登录状态
        String token = request.getHeader("authorization");
        if (StrUtil.isBlank(token)) {
            response.setStatus(401);
            return false;
        }
        
        // 3. 验证token
        UserDTO user = userService.getUserByToken(token);
        if (user == null) {
            response.setStatus(401);
            return false;
        }
        
        // 4. 保存用户信息到ThreadLocal
        UserHolder.saveUser(user);
        return true;
    }
    
    @Override
    public void afterCompletion(HttpServletRequest request, 
                               HttpServletResponse response, 
                               Object handler, Exception ex) {
        // 清理ThreadLocal，防止内存泄漏
        UserHolder.removeUser();
    }
}

// 配置拦截器
@Configuration
public class MvcConfig implements WebMvcConfigurer {
    @Override
    public void addInterceptors(InterceptorRegistry registry) {
        registry.addInterceptor(new LoginInterceptor())
                .excludePathPatterns("/user/login", "/user/code")
                .order(1);
    }
}
```

**为什么选择Interceptor：**
1. 粒度更细：可精确到Controller方法
2. 可获取HandlerMethod：可判断方法上的注解
3. 可处理异常：afterCompletion可获取异常信息
4. 依赖注入方便：可直接使用Spring Bean

```java
                      @Header(AmqpHeaders.DELIVERY_TAG) long tag) {
    try {
        // 1. 处理日志
        auditLogMapper.insert(log);
        
        // 2. 手动ACK（确认消费成功）
        channel.basicAck(tag, false);

        channel.basicNack(tag, false, true); // requeue=true，重新入队

        }
    }
```

### 题目7：缓存击穿解决方案

**原题：** 标准的互斥锁（Mutex Lock）和逻辑过期（Logical Expire）解决击穿有什么区别？为什么在你的场景下选择了逻辑过期？它的优缺点是什么？

**你的回答：** 互斥锁是缓存过期之后就去重建缓存，只有一个线程能拿到锁去异步查询重建，而逻辑过期是缓存数据部显式指定过期时间，而是数据内存储一个逻辑过期时间，查询过期之后一个线程获得锁去异步重建缓存，并返回旧值。采用逻辑过期是因为对实时一致性要求不高，所以返回旧值不阻塞提升用户体验。

**评判：**
- 优点：基本理解了两种方案的区别
- 不足：未说明互斥锁会阻塞用户请求，未说明逻辑过期的缺点

**标准答案：**

**两种方案对比：**

| 特性             | 互斥锁（Mutex Lock）             | 逻辑过期（Logical Expire）             |
| ---------------- | -------------------------------- | -------------------------------------- |
| **缓存过期时机** | Redis TTL过期，缓存被删除        | 缓存永不过期，逻辑过期时间存储在数据中 |
| **用户请求处理** | 缓存过期后，所有请求阻塞等待重建 | 返回旧数据，后台异步重建               |
| **数据一致性**   | 强一致性（返回最新数据）         | 最终一致性（可能返回过期数据）         |
| **用户体验**     | 可能阻塞，响应慢                 | 不阻塞，响应快                         |
| **实现复杂度**   | 简单                             | 较复杂                                 |

**代码实现：**

```java
// 方案1：互斥锁
public Shop queryWithMutex(Long id) {
    String key = CACHE_SHOP_KEY + id;
    // 1. 查询缓存
    String shopJson = stringRedisTemplate.opsForValue().get(key);
    if (StrUtil.isNotBlank(shopJson)) {
        return JSONUtil.toBean(shopJson, Shop.class);
    }
    
    // 2. 缓存未命中，尝试获取锁
    String lockKey = LOCK_SHOP_KEY + id;
    boolean isLock = tryLock(lockKey);
    
    if (!isLock) {
        // 3. 获取锁失败，休眠后重试
        Thread.sleep(50);
        return queryWithMutex(id); // 递归重试
    }
    
    try {
        // 4. 获取锁成功，双重检查
        shopJson = stringRedisTemplate.opsForValue().get(key);
        if (StrUtil.isNotBlank(shopJson)) {
            return JSONUtil.toBean(shopJson, Shop.class);
        }
        
        // 5. 查询数据库
        Shop shop = shopMapper.selectById(id);
        if (shop == null) {
            // 缓存空值，防止缓存穿透
            stringRedisTemplate.opsForValue().set(key, "", 2, TimeUnit.MINUTES);
            return null;
        }
        
        // 6. 写入缓存
        stringRedisTemplate.opsForValue().set(key, JSONUtil.toJsonStr(shop), 30, TimeUnit.MINUTES);
        return shop;
    } finally {
        unlock(lockKey);
    }
}

// 方案2：逻辑过期
public Shop queryWithLogicalExpire(Long id) {
    String key = CACHE_SHOP_KEY + id;
    // 1. 查询缓存
    String json = stringRedisTemplate.opsForValue().get(key);
    if (StrUtil.isBlank(json)) {
        // 缓存未命中，返回null（需要预热）
        return null;
    }
    
    // 2. 反序列化
    RedisData redisData = JSONUtil.toBean(json, RedisData.class);
    Shop shop = JSONUtil.toBean((JSONObject) redisData.getData(), Shop.class);
    LocalDateTime expireTime = redisData.getExpireTime();
    
    // 3. 判断是否过期
    if (expireTime.isAfter(LocalDateTime.now())) {
        // 未过期，直接返回
        return shop;
    }
    
    // 4. 已过期，尝试获取锁重建
    String lockKey = LOCK_SHOP_KEY + id;
    boolean isLock = tryLock(lockKey);
    
    if (isLock) {
        // 5. 获取锁成功，开启独立线程重建
        CACHE_REBUILD_EXECUTOR.submit(() -> {
            try {
                // 查询数据库
                Shop newShop = shopMapper.selectById(id);
                // 重建缓存
                setWithLogicalExpire(key, newShop, 20L, TimeUnit.SECONDS);
            } finally {
                unlock(lockKey);
            }
        });
    }
    
    // 6. 返回旧数据（无论是否获取到锁）
    return shop;
}
```

**为什么选择逻辑过期：**
1. 用户体验好：不阻塞，响应快
2. 高并发场景：大量请求不会同时阻塞
3. 数据容忍度：店铺信息更新不频繁，可接受短暂过期

**逻辑过期的缺点：**
1. 数据可能过期：返回旧数据，需要业务容忍
2. 需要预热：首次查询缓存未命中
3. 内存占用：缓存永不过期，占用内存

### 题目9：Redlock问题

**原题：** Redis 如果是主从架构，主节点挂了，锁还没同步到从节点，这时候从节点变主节点，另一个线程来加锁也成功了，这就导致了锁失效（Redlock 问题）。你在项目里考虑过这个问题吗？怎么解决的？

**你的回答：** 没有设计，可以考虑红锁。

**评判：**
- 优点：知道Redlock
- 不足：未说明Redlock的具体实现和适用场景

**标准答案：**

**问题分析：**
1. 主从异步复制：主节点写入锁后，从节点可能未同步
2. 主节点故障：从节点晋升为主节点，锁丢失
3. 锁失效：另一个线程可获取同一把锁

**解决方案：**

```java
// 方案1：Redlock算法（适合高可靠性场景）

// 3. 锁的过期时间 = 锁的有效时间 - 获取锁耗时
// 4. 释放锁时，向所有节点发送释放请求

// 方案2：单Redis + 业务层保证（适合一般场景）
// 1. 锁的过期时间设置合理（大于业务执行时间）
// 2. 看门狗机制自动续期
// 3. 业务幂等性保证

// 方案3：Zookeeper分布式锁（强一致性）
// Zookeeper的写操作需要大多数节点确认，天然支持Redlock
```

**Redlock算法步骤：**
1. 获取当前时间T1
2. 依次向N个Redis节点请求加锁
3. 计算加锁耗时 = 当前时间T2 - T1
4. 锁的有效时间 = 锁的过期时间 - 加锁耗时
5. 如果大多数节点加锁成功 && 锁的有效时间 > 0，则加锁成功
6. 否则，向所有节点发送释放锁请求

**适用场景：**
- Redlock：金融、支付等强一致性场景
- 单Redis + 看门狗：一般业务场景（如秒杀）

### 题目10：看门狗机制

**原题：** 如果你的业务代码在执行业务时，服务器突然断电了，锁没释放怎么办？Redisson 的看门狗（Watch Dog）机制是怎么工作的？

**你的回答：** 断电了锁信息一起持久化，回复了应该就可以把。看门狗是一个后台线程每隔一段时间检查线程释放还持有锁，如果持有就延迟锁过期时间。

**评判：**
- 优点：基本理解了看门狗机制
- 不足：断电后锁信息不会持久化（Redis是内存数据库），未说明看门狗的续期逻辑

**标准答案：**

**问题分析：**
1. 服务器断电：JVM进程终止，锁无法释放
2. Redis锁过期：锁设置了TTL，过期后自动释放
3. 但业务可能还在执行：如果业务执行时间 > 锁的TTL，锁提前释放

**看门狗机制：**

```java
    }
    
    // 看门狗续期逻辑
    private void scheduleExpirationRenewal(String threadId) {
        // 1. 创建定时任务，每 lockWatchdogTimeout / 3 秒执行一次
        Timeout task = commandExecutor.getConnectionManager()
            .newTimeout(new TimerTask() {
                @Override
                public void run(Timeout timeout) {
                    // 2. 检查当前线程是否还持有锁
                    if (internalLockLeaseTime == -1) { // -1表示启用看门狗
                        // 3. 续期：重新设置锁的过期时间
                        RFuture<Boolean> future = renewExpirationAsync(threadId);
                        future.onComplete((res, e) -> {
                            if (e != null) {
                                // 续期失败，停止看门狗
                                return;
                            }
                            if (res) {
                                // 续期成功，继续下一次续期
                                scheduleExpirationRenewal(threadId);
                            }
                        });
                    }
                }
            }, lockWatchdogTimeout / 3, TimeUnit.MILLISECONDS); // 默认10秒续期一次
    }
    
    // 续期Lua脚本
    String renewScript = 
        "if (redis.call('hexists', KEYS[1], ARGV[2]) == 1) then " +
        "  redis.call('pexpire', KEYS[1], ARGV[1]); " +  // 重新设置过期时间
        "  return 1; " +
        "end; " +
        "return 0;";
}
```

**看门狗工作流程：**
1. 获取锁时不指定leaseTime（或leaseTime=-1），启用看门狗
2. 看门狗每10秒（lockWatchdogTimeout/3）检查一次
3. 如果线程还持有锁，执行续期（重新设置30秒过期时间）
4. 如果线程释放锁或进程终止，看门狗停止续期
5. 锁在30秒后自动过期释放

**断电场景处理：**
1. 服务器断电：JVM进程终止，看门狗线程停止
2. 锁自动过期：Redis中的锁TTL到期后自动删除
3. 业务可能未完成：如果业务执行时间 > 30秒，锁提前释放，可能导致并发问题

**最佳实践：**
1. 合理设置锁的过期时间：大于业务执行时间
2. 启用看门狗：自动续期，防止业务未完成锁就过期
3. 业务幂等性：即使锁提前释放，业务也要保证幂等

### 题目12：Redis Stream消息重试

**原题：** 消费者拿到消息后，处理业务失败了，Redis Stream 是怎么保证消息重试的？你是手动 ACK 还是自动 ACK？

**你的回答：** 手动ACK，保证重试的话是检查pendinglist有消息就继续消费直到ACK。

**评判：**
- 优点：知道手动ACK和pending-list
- 不足：未说明pending-list的具体处理逻辑

**标准答案：**

**消息重试机制：**

```java
// 手动ACK + Pending-List处理
private class VoucherOrderHandler implements Runnable {
    @Override
    public void run() {
        while (true) {
            try {
                // 1. 先处理pending-list（失败重试的消息）
                handlePendingList();
                
                // 2. 处理新消息
                List<MapRecord<String, Object, Object>> list = 
                    stringRedisTemplate.opsForStream().read(
                        Consumer.from("g1", "c1"),
                        StreamReadOptions.empty()
                            .count(1)
                            .block(Duration.ofSeconds(2)),
                        StreamOffset.create("stream.orders", ReadOffset.lastConsumed())
                    );
                
                if (list == null || list.isEmpty()) {
                    continue;
                }
                
                // 3. 处理消息
                MapRecord<String, Object, Object> record = list.get(0);
                processMessage(record);
                
            } catch (Exception e) {
                log.error("处理订单异常", e);
            }
        }
    }
    
    // 处理pending-list（失败重试）
    private void handlePendingList() {
        while (true) {
            try {
                // XREADGROUP GROUP g1 c1 COUNT 1 STREAMS stream.orders 0
                // 0表示从pending-list读取
                List<MapRecord<String, Object, Object>> list = 
                    stringRedisTemplate.opsForStream().read(
                        Consumer.from("g1", "c1"),
                        StreamReadOptions.empty().count(1),
                        StreamOffset.create("stream.orders", ReadOffset.from("0"))
                    );
                
                if (list == null || list.isEmpty()) {
                    break; // pending-list为空，退出
                }
                
                // 处理pending消息
                MapRecord<String, Object, Object> record = list.get(0);
                processMessage(record);
                
            } catch (Exception e) {
                log.error("处理pending消息异常", e);
                break;
            }
        }
    }
    
    // 处理消息（带重试机制）
    private void processMessage(MapRecord<String, Object, Object> record) {
        int retryCount = 0;
        int maxRetries = 3;
        
        while (retryCount < maxRetries) {
            try {
                // 1. 解析消息
                VoucherOrder order = parseOrder(record);
                
                // 2. 处理业务
                createVoucherOrder(order);
                
                // 3. 手动ACK（确认消费成功）
                stringRedisTemplate.opsForStream().acknowledge(
                    "stream.orders", "g1", record.getId()
                );
                
                return; // 处理成功，退出
                
            } catch (Exception e) {
                retryCount++;
                log.error("处理消息失败，重试次数：{}/{}", retryCount, maxRetries, e);
                
                if (retryCount >= maxRetries) {
                    // 超过重试次数，发送到死信队列或记录日志
                    sendToDeadLetterQueue(record);
                    // 仍然ACK，避免无限重试
                    stringRedisTemplate.opsForStream().acknowledge(
                        "stream.orders", "g1", record.getId()
                    );
                } else {
                    // 等待后重试
                    try {
                        Thread.sleep(1000 * retryCount); // 指数退避
                    } catch (InterruptedException ie) {
                        Thread.currentThread().interrupt();
                    }
                }
            }
        }
    }
}
```

**Redis Stream消息流转：**
1. 新消息：`XADD stream.orders * field value` → 进入stream
2. 消费者读取：`XREADGROUP GROUP g1 c1 STREAMS stream.orders >` → 消息进入pending-list
3. 处理成功：`XACK stream.orders g1 messageId` → 从pending-list移除
4. 处理失败：不ACK → 消息留在pending-list
5. 重试：`XREADGROUP GROUP g1 c1 STREAMS stream.orders 0` → 从pending-list读取

**手动ACK vs 自动ACK：**
- 手动ACK：处理成功后才ACK，失败消息留在pending-list，可重试
- 自动ACK：读取即ACK，失败消息无法重试（不推荐）

**原题：** 你提到了 "用户维度的分布式锁"，具体的 Key 是怎么设计的？如果有一个用户疯狂点击，锁会不会一直被他自己重入持有，导致其他用户也买不到？

**你的回答：** 前缀+userid。不会，锁的是用户维度，不同用户是并行的。

**评判：**
- 优点：正确理解了用户维度锁
- 不足：未说明重入锁的问题，未说明如何防止同一用户重复下单

**标准答案：**

**锁的Key设计：**

```java
// 锁的Key：lock:order:{userId}
String lockKey = "lock:order:" + userId;
RLock lock = redissonClient.getLock(lockKey);

// 为什么用userId而不是voucherId？
// 1. 一人一单：限制的是用户维度，不是商品维度
// 2. 不同用户并行：不同用户的锁不同，不会相互阻塞
// 3. 同一用户串行：同一用户的多次请求串行处理，防止重复下单
```

**重入锁问题分析：**

```java
// 场景：用户疯狂点击
// 请求1：获取锁 lock:order:123，执行业务（查询订单、扣库存、创建订单）
// 请求2：同一用户，获取同一把锁 lock:order:123，等待请求1释放
// 请求3：同一用户，获取同一把锁 lock:order:123，等待请求1释放

// 问题：如果使用可重入锁，请求2和请求3会重入，导致重复下单？

// 答案：不会，因为业务逻辑中有重复下单检查
private void createVoucherOrder(VoucherOrder voucherOrder) {
    Long userId = voucherOrder.getUserId();
    Long voucherId = voucherOrder.getVoucherId();
    
    RLock lock = redissonClient.getLock("lock:order:" + userId);
    boolean isLock = lock.tryLock();  // 非阻塞获取锁
    
    if (!isLock) {
        // 获取锁失败，说明该用户正在处理订单，直接返回
        log.error("不允许重复下单！");
        return;
    }
    
    try {
        // 1. 查询订单（双重检查）
        int count = query().eq("user_id", userId)
                          .eq("voucher_id", voucherId)
                          .count();
        if (count > 0) {
            // 用户已经购买过了
            return;
        }
        
        // 2. 扣减库存
        boolean success = seckillVoucherService.update()
            .setSql("stock = stock - 1")
            .eq("voucher_id", voucherId)
            .gt("stock", 0)
            .update();
        
        if (!success) {
            return;
        }
        
        // 3. 创建订单
        save(voucherOrder);
        
    } finally {
        lock.unlock();
    }
}
```

**为什么不会导致其他用户买不到：**
1. 锁的粒度是用户维度：`lock:order:{userId}`
2. 不同用户的锁不同：用户A的锁不影响用户B
3. 同一用户的请求串行：防止重复下单

**防止同一用户重复下单的措施：**
1. 分布式锁：同一用户的请求串行处理
2. 数据库唯一索引：`UNIQUE KEY uk_user_voucher (user_id, voucher_id)`
3. Redis Set记录：`SADD seckill:order:{voucherId} {userId}`

### 题目15：ThreadPoolExecutor参数

**原题：** 如果让你手写一个ThreadPoolExecutor，那 7 个参数分别是什么？workQueue（工作队列）满了之后，线程池会怎么处理新来的任务？（拒绝策略）

**你的回答：** 核心线程数，最大线程数，非核心线程存活时间，时间单位，工作队列，解决策略，线程池工厂。拒绝策略有callerrunpolicy\abort\abort latest\ajected。

**评判：**
- 优点：基本说出了7个参数
- 不足：参数名称不够准确，拒绝策略名称有误

**标准答案：**

**ThreadPoolExecutor的7个参数：**

```java
public ThreadPoolExecutor(
    int corePoolSize,              // 1. 核心线程数
    int maximumPoolSize,           // 2. 最大线程数
    long keepAliveTime,            // 3. 非核心线程存活时间
    TimeUnit unit,                 // 4. 时间单位
    BlockingQueue<Runnable> workQueue,  // 5. 工作队列
    ThreadFactory threadFactory,   // 6. 线程工厂
    RejectedExecutionHandler handler    // 7. 拒绝策略
)
```

**参数详解：**
1. **corePoolSize**：核心线程数，线程池长期保持的线程数量
2. **maximumPoolSize**：最大线程数，线程池允许的最大线程数量
3. **keepAliveTime**：非核心线程空闲时的最大存活时间
4. **unit**：时间单位（TimeUnit.SECONDS等）
5. **workQueue**：工作队列，存储待执行的任务
6. **threadFactory**：线程工厂，用于创建线程
7. **handler**：拒绝策略，当队列满且线程数达到最大值时的处理策略

**拒绝策略（4种）：**

```java
// 1. AbortPolicy（默认）：抛出异常
new ThreadPoolExecutor.AbortPolicy();
// 抛出 RejectedExecutionException

// 2. CallerRunsPolicy：由调用线程执行
new ThreadPoolExecutor.CallerRunsPolicy();
// 如果线程池未关闭，由提交任务的线程执行

// 3. DiscardPolicy：静默丢弃
new ThreadPoolExecutor.DiscardPolicy();
// 什么都不做，静默丢弃任务

// 4. DiscardOldestPolicy：丢弃最老的任务
new ThreadPoolExecutor.DiscardOldestPolicy();
// 丢弃队列中最老的任务，然后重新提交当前任务
```

**执行流程：**
```
1. 提交任务
   ↓
2. 当前线程数 < corePoolSize？
   ├─ 是 → 创建核心线程执行
   └─ 否 → 3
   ↓
3. 任务入队（workQueue.offer()）
   ├─ 成功 → 等待执行
   └─ 失败（队列满）→ 4
   ↓
4. 当前线程数 < maximumPoolSize？
   ├─ 是 → 创建非核心线程执行
   └─ 否 → 5
   ↓
5. 执行拒绝策略
```

### 题目16：Full GC频繁排查

1. 内存泄漏：对象无法被回收，老年代逐渐填满

**排查步骤：**

```bash
# 3. 导出堆转储文件
jmap -dump:format=b,file=heap.hprof <pid>

# 4. 查看对象统计
jmap -histo <pid> | head -20

# 5. 实时监控GC
jstat -gc <pid> 1000 10  # 每1秒输出一次，共10次
```

**使用MAT分析堆转储：**
1. 打开MAT，加载heap.hprof
2. 查看Histogram：找出占用内存大的对象
3. 查看Dominator Tree：找出持有大量内存的对象
4. 查看Leak Suspects：自动检测内存泄漏

### 题目17：MySQL索引

**原题：** 在你的医美系统里，"入库单标识" 需要支持筛选，你在设计表结构时，这个字段加索引了吗？如果这个字段是字符串且很长，用什么索引类型（Hash 还是 B+Tree）？为什么？

**你的回答：** 这个是帅选字段，加了索引。很长的话考虑hash索引。

**评判：**
- 优点：知道需要加索引
- 不足：回答有误，MySQL的Hash索引只支持Memory引擎，InnoDB不支持Hash索引

**标准答案：**

**索引设计：**

```sql
-- 入库单表
CREATE TABLE `inbound_order` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `order_no` VARCHAR(64) NOT NULL COMMENT '入库单标识',
  `comp_id` CHAR(36) NOT NULL,
  `create_time` DATETIME NOT NULL,
  -- 其他字段...
  INDEX `idx_order_no` (`order_no`)  -- 普通B+Tree索引
) ENGINE=InnoDB;
```

**Hash索引 vs B+Tree索引：**

| 特性           | Hash索引         | B+Tree索引       |
| -------------- | ---------------- | ---------------- |
| **支持引擎**   | 仅Memory引擎     | InnoDB、MyISAM等 |
| **查询类型**   | 仅等值查询（=）  | 等值、范围、排序 |
| **时间复杂度** | O(1)             | O(log n)         |
| **字符串长度** | 无限制           | 无限制           |
| **适用场景**   | 内存表，等值查询 | 大多数场景       |

**为什么选择B+Tree索引：**
1. InnoDB不支持Hash索引：MySQL的Hash索引仅Memory引擎支持
2. 支持范围查询：`WHERE order_no > 'xxx'`
3. 支持排序：`ORDER BY order_no`
4. 支持前缀索引：`INDEX idx_order_no (order_no(10))`（如果字符串很长）

**字符串很长时的优化：**

```sql
-- 方案1：前缀索引（推荐）
CREATE INDEX idx_order_no ON inbound_order(order_no(20));
-- 只索引前20个字符，节省空间
```

### 题目18：@Transactional事务失效

**原题：** 如果一个入库方法 A 调用了同一个类里的另一个有@Transactional注解的方法 B，事务会生效吗？为什么？怎么解决？

**你的回答：** 不会。

**评判：**
- 优点：知道事务不会生效
- 不足：未说明原因和解决方案

**标准答案：**

**问题分析：**

```java
@Service
public class InboundService {
    
    public void methodA() {
        // 方法A没有事务
        this.methodB();  // 调用同一个类的方法B
    }
    
    @Transactional
    public void methodB() {
        // 方法B有事务注解，但不会生效
        inboundMapper.insert(order);
    }
}
```

**为什么不会生效：**
1. Spring事务基于AOP代理：`@Transactional`通过代理对象生效
2. 内部调用绕过代理：`this.methodB()`是直接调用，不是代理对象调用
3. 事务不生效：AOP拦截不到内部调用

**解决方案：**

```java
// 方案1：将方法B提取到另一个Service（推荐）
@Service
public class InboundService {
    @Autowired
    private InboundDetailService inboundDetailService;
    
    public void methodA() {
        inboundDetailService.methodB();  // 通过代理对象调用
    }
}

@Service
public class InboundDetailService {
    @Transactional
    public void methodB() {
        // 事务生效
    }
}

// 方案2：方法A也加@Transactional
@Service
public class InboundService {
    @Transactional
    public void methodA() {
        this.methodB();  // 在同一个事务中
    }
    
    public void methodB() {
        // 不需要@Transactional，已经在事务中
    }
}

// 方案3：通过ApplicationContext获取代理对象
@Service
public class InboundService implements ApplicationContextAware {
    private ApplicationContext applicationContext;
    
    @Override
    public void setApplicationContext(ApplicationContext applicationContext) {
        this.applicationContext = applicationContext;
    }
    
    public void methodA() {
        InboundService proxy = applicationContext.getBean(InboundService.class);
        proxy.methodB();  // 通过代理对象调用
    }
    
    @Transactional
    public void methodB() {
        // 事务生效
    }
}

// 方案4：使用AopContext（需要开启exposeProxy）
@EnableAspectJAutoProxy(exposeProxy = true)
@Service
public class InboundService {
    public void methodA() {
        InboundService proxy = (InboundService) AopContext.currentProxy();
        proxy.methodB();  // 通过代理对象调用
    }
    
    @Transactional
    public void methodB() {
        // 事务生效
    }
}
```

**推荐方案：** 方案1（提取到另一个Service），代码清晰，符合单一职责原则

```java
// 计算距离（Haversine公式）
private double calculateDistance(double lng1, double lat1, 
                                 double lng2, double lat2) {
    double R = 6371; // 地球半径（公里）
    double dLat = Math.toRadians(lat2 - lat1);
    double dLng = Math.toRadians(lng2 - lng1);
    double a = Math.sin(dLat / 2) * Math.sin(dLat / 2) +
               Math.cos(Math.toRadians(lat1)) * 
               Math.cos(Math.toRadians(lat2)) *
               Math.sin(dLng / 2) * Math.sin(dLng / 2);
    double c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
    return R * c;
}

// 方案2：K-D Tree（适合数据量大，>1万）
public class KDTree {
    private Node root;
    
    // 构建K-D Tree
    public void build(List<Point> points) {
        root = buildTree(points, 0);
    }
    
    // 查询最近的K个点
    public List<Point> findKNearest(Point target, int k) {
        PriorityQueue<PointDistance> heap = new PriorityQueue<>(
            (a, b) -> Double.compare(b.distance, a.distance)
        );
        search(root, target, k, heap, 0);
        return heap.stream()
            .map(pd -> pd.point)
            .collect(Collectors.toList());
    }
    
    private void search(Node node, Point target, int k, 
                       PriorityQueue<PointDistance> heap, int depth) {
        if (node == null) return;
        
        // 计算距离
        double distance = calculateDistance(node.point, target);
        if (heap.size() < k || distance < heap.peek().distance) {
            heap.offer(new PointDistance(node.point, distance));
            if (heap.size() > k) {
                heap.poll();
            }
        }
        
        // 选择搜索方向
        int axis = depth % 2;
        double diff = axis == 0 ? 
            target.x - node.point.x : 
            target.y - node.point.y;
        
        Node near = diff < 0 ? node.left : node.right;
        Node far = diff < 0 ? node.right : node.left;
        
        search(near, target, k, heap, depth + 1);
        
        // 如果最远距离 > 分割线距离，需要搜索另一侧
        if (heap.size() < k || 
            Math.abs(diff) < heap.peek().distance) {
            search(far, target, k, heap, depth + 1);
        }
    }
}
```
