**前端与AI交互的核心价值**：

- **用户体验革新**：从静态界面到智能交互，提供个性化、实时的响应
- **功能边界扩展**：实现自然语言理解、图像生成、语音交互等复杂能力

### 1.1.2 通信模式对比

| 通信模式 | 技术方案 | 适用场景 | 优点 | 缺点 |
|----------|----------|----------|------|------|
| **REST API** | HTTP POST/GET | 一次性请求-响应 | 简单易用，广泛支持 | 实时性差，不适合长任务 |
| **WebSocket** | 全双工WebSocket | 实时双向通信 | 低延迟，支持双向通信 | 实现复杂，需管理连接状态 |
| **Server-Sent Events** | SSE | 服务器单向推送 | 实现简单，适合流式输出 | 只支持单向通信 |
| **本地推理** | WebAssembly/ONNX Runtime | 浏览器内运行 | 隐私性好，无网络依赖 | 模型大小受限，性能要求高 |
| **混合模式** | 本地+云端 | 智能场景切换 | 兼顾性能和功能 | 架构复杂，需协调策略 |

```javascript
// api.js - 基础API调用封装
import axios from 'axios';

class AIApiClient {
  constructor(apiKey, baseUrl = 'https://api.openai.com/v1') {
    this.client = axios.create({
      baseURL: baseUrl,
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${apiKey}`
      },
      timeout: 30000 // 30秒超时
    });
    
    // 请求拦截器
    this.client.interceptors.request.use(
      (config) => {
        console.log('AI API Request:', config.url);
        return config;
      },
      (error) => {
        return Promise.reject(error);
      }
    );
    
    // 响应拦截器
    this.client.interceptors.response.use(
      (response) => {
        return response.data;
      },
      (error) => {
        if (error.response) {
          // 服务器返回错误
          console.error('AI API Error:', error.response.data);
        } else if (error.request) {
          // 请求发送但无响应
          console.error('AI API Network Error:', error.request);
        } else {
          // 请求配置错误
          console.error('AI API Config Error:', error.message);
        }
        return Promise.reject(error);
      }
    );
  }
  
  // 聊天完成
  async chatComplete(messages, model = 'gpt-3.5-turbo') {
    return this.client.post('/chat/completions', {
      model,
      messages,
      temperature: 0.7,
      max_tokens: 1000
    });
  }
  
  // 文本完成
  async textComplete(prompt, model = 'text-davinci-003') {
    return this.client.post('/completions', {
      model,
      prompt,
      temperature: 0.7,
      max_tokens: 500
    });
  }
  
  // 嵌入生成
  async createEmbedding(input, model = 'text-embedding-ada-002') {
    return this.client.post('/embeddings', {
      model,
      input
    });
  }
}
```

### 1.2.2 WebSocket流式通信

```javascript
// websocketService.js - WebSocket服务
class WebSocketService {
  constructor(url) {
    this.url = url;
    this.socket = null;
    this.reconnectAttempts = 0;
    this.maxReconnectAttempts = 5;
    this.reconnectDelay = 1000;
    this.listeners = {};
  }
  
  connect() {
    return new Promise((resolve, reject) => {
      try {
        this.socket = new WebSocket(this.url);
        
        this.socket.onopen = () => {
          console.log('WebSocket connected');
          this.reconnectAttempts = 0;
          resolve();
        };
        
        this.socket.onmessage = (event) => {
          this._handleMessage(event.data);
        };
        
        this.socket.onclose = () => {
          console.log('WebSocket disconnected');
          this._reconnect();
        };
        
        this.socket.onerror = (error) => {
          console.error('WebSocket error:', error);
          reject(error);
        };
      } catch (error) {
        console.error('WebSocket connection failed:', error);
        reject(error);
      }
    });
  }
  
  send(data) {
    if (this.socket && this.socket.readyState === WebSocket.OPEN) {
      this.socket.send(JSON.stringify(data));
      return true;
    }
    console.warn('WebSocket not connected');
    return false;
  }
  
  close() {
    if (this.socket) {
      this.socket.close();
      this.socket = null;
    }
  }
  
  on(event, callback) {
    if (!this.listeners[event]) {
      this.listeners[event] = [];
    }
    this.listeners[event].push(callback);
  }
  
  off(event, callback) {
    if (this.listeners[event]) {
      this.listeners[event] = this.listeners[event].filter(cb => cb !== callback);
    }
  }
  
  _handleMessage(data) {
    try {
      const parsedData = JSON.parse(data);
      if (parsedData.type) {
        const callbacks = this.listeners[parsedData.type] || [];
        callbacks.forEach(callback => callback(parsedData));
      }
    } catch (error) {
      console.error('Failed to parse WebSocket message:', error);
    }
  }
  
  _reconnect() {
    if (this.reconnectAttempts < this.maxReconnectAttempts) {
      this.reconnectAttempts++;
      const delay = this.reconnectDelay * Math.pow(2, this.reconnectAttempts - 1);
      
      console.log(`Attempting to reconnect in ${delay}ms...`);
      
      setTimeout(() => {
        console.log(`Reconnect attempt ${this.reconnectAttempts}/${this.maxReconnectAttempts}`);
        this.connect().catch(error => {
          console.error('Reconnect failed:', error);
        });
      }, delay);
    } else {
      console.error('Max reconnect attempts reached');
    }
  }
}

// 使用示例
const wsService = new WebSocketService('wss://api.example.com/ai-stream');

wsService.on('message', (data) => {
  console.log('Received message:', data.content);
});

wsService.on('error', (data) => {
  console.error('Stream error:', data.error);
});

wsService.on('end', () => {
  console.log('Stream ended');
});

async function startStream() {
  try {
    await wsService.connect();
    wsService.send({
      type: 'chat',
      messages: [
        { role: 'user', content: 'Tell me a story about AI' }
      ],
      model: 'gpt-4'
    });
  } catch (error) {
    console.error('Failed to start stream:', error);
  }
}
```

### 1.2.3 本地推理实现

```javascript
// localInference.js - 浏览器内AI推理
import * as tf from '@tensorflow/tfjs';
import { AutoModel, AutoTokenizer } from '@xenova/transformers';

class LocalInference {
  constructor() {
    this.model = null;
    this.tokenizer = null;
    this.isLoading = false;
  }
  
  async loadModel(modelName = 'Xenova/distilbert-base-uncased-finetuned-sst-2-english') {
    if (this.isLoading) {
      console.warn('Model is already loading');
      return;
    }
    
    this.isLoading = true;
    
    try {
      console.log(`Loading model: ${modelName}`);
      
      // 加载分词器
      this.tokenizer = await AutoTokenizer.from_pretrained(modelName);
      
      // 加载模型
      this.model = await AutoModel.from_pretrained(modelName);
      
      console.log('Model loaded successfully');
    } catch (error) {
      console.error('Failed to load model:', error);
      throw error;
    } finally {
      this.isLoading = false;
    }
  }
  
  async infer(text) {
    if (!this.model || !this.tokenizer) {
      throw new Error('Model not loaded');
    }
    
    try {
      // 分词
      const inputs = this.tokenizer(text, { return_tensors: 'tf' });
      
      // 推理
      const outputs = await this.model(inputs);
      
      // 处理输出
      const logits = outputs.logits;
      const probabilities = tf.softmax(logits).dataSync();
      
      // 转换为结果
      const labels = ['negative', 'positive'];
      const predictedIndex = tf.argMax(logits, 1).dataSync()[0];
      const predictedLabel = labels[predictedIndex];
      const confidence = probabilities[predictedIndex];
      
      return {
        label: predictedLabel,
        confidence: confidence
      };
    } catch (error) {
      console.error('Inference failed:', error);
      throw error;
    }
  }
  
  async batchInfer(texts) {
    const results = [];
    
    for (const text of texts) {
      try {
        const result = await this.infer(text);
        results.push({ text, ...result });
      } catch (error) {
        results.push({ text, error: error.message });
      }
    }
    
    return results;
  }
  
  dispose() {
    if (this.model) {
      this.model.dispose();
      this.model = null;
    }
    if (this.tokenizer) {
      this.tokenizer = null;
    }
    console.log('Model disposed');
  }
}

// 使用示例
const localInference = new LocalInference();

async function sentimentAnalysis() {
  try {
    await localInference.loadModel();
    
    const result = await localInference.infer('I love AI frontend development!');
    console.log('Sentiment analysis result:', result);
    
    const batchResults = await localInference.batchInfer([
      'The UI is amazing',
      'The performance is terrible',
      'I enjoy working with this library'
    ]);
    console.log('Batch results:', batchResults);
  } catch (error) {
    console.error('Error:', error);
  } finally {
    localInference.dispose();
  }
}
```

### 1.3.2 性能优化策略

**网络优化**：
- 使用HTTP/2或HTTP/3减少连接开销
- 实现请求批处理，减少API调用次数
- 使用CDN加速静态资源加载
- 实现请求缓存，避免重复请求

**渲染优化**：
- 使用虚拟列表处理长对话历史
- 实现流式渲染，逐字显示生成内容
- 使用防抖和节流处理用户输入
- 采用代码分割，按需加载组件

**状态管理优化**：
- 合理设计状态结构，避免过度渲染
- 使用持久化存储，保持会话状态
- 实现状态缓存，提高响应速度
- 采用乐观更新，提升用户体验

### 1.3.3 安全最佳实践

**API密钥管理**：
- 避免在前端硬编码API密钥
- 使用环境变量或后端代理
- 实现API密钥轮换机制
- 对敏感操作进行后端验证

**数据安全**：
- 使用HTTPS加密传输
- 实现输入验证，防止注入攻击
- 对敏感数据进行加密存储
- 遵守数据隐私法规（GDPR、CCPA等）

**网络安全**：
- 正确配置CORS策略
- 实现请求速率限制
- 防范XSS和CSRF攻击
- 使用内容安全策略（CSP）

**状态管理架构**：

```javascript
// store.js - Zustand状态管理示例
import create from 'zustand';
import { persist } from 'zustand/middleware';

const useAIStore = create(
  persist(
    (set, get) => ({
      // 对话状态
      conversations: [],
      currentConversationId: null,
      
      // 消息状态
      messages: {}, // conversationId -> messages
      
      // 生成状态
      isGenerating: false,
      generationProgress: 0,
      
      // 错误状态
      error: null,
      
      // 配置状态
      model: 'gpt-3.5-turbo',
      temperature: 0.7,
      maxTokens: 1000,
      
      // 操作方法
      addConversation: (title) => {
        const newConversation = {
          id: Date.now().toString(),
          title,
          createdAt: new Date().toISOString(),
          updatedAt: new Date().toISOString()
        };
        
        set((state) => ({
          conversations: [...state.conversations, newConversation],
          currentConversationId: newConversation.id,
          messages: {
            ...state.messages,
            [newConversation.id]: []
          }
        }));
        
        return newConversation.id;
      },
      
      addMessage: (conversationId, message) => {
        set((state) => ({
          messages: {
            ...state.messages,
            [conversationId]: [...(state.messages[conversationId] || []), message]
          },
          conversations: state.conversations.map(conv => 
            conv.id === conversationId 
              ? { ...conv, updatedAt: new Date().toISOString() }
              : conv
          )
        }));
      },
      
      updateMessage: (conversationId, messageId, updates) => {
        set((state) => ({
          messages: {
            ...state.messages,
            [conversationId]: state.messages[conversationId].map(msg => 
              msg.id === messageId ? { ...msg, ...updates } : msg
            )
          }
        }));
      },
      
      setGenerating: (isGenerating) => set({ isGenerating }),
      
      setGenerationProgress: (progress) => set({ generationProgress: Math.min(progress, 100) }),
      
      setError: (error) => set({ error }),
      
      setModel: (model) => set({ model }),
      
      setTemperature: (temperature) => set({ temperature }),
      
      setMaxTokens: (maxTokens) => set({ maxTokens }),
      
      deleteConversation: (conversationId) => {
        set((state) => {
          const newConversations = state.conversations.filter(
            conv => conv.id !== conversationId
          );
          
          const newMessages = { ...state.messages };
          delete newMessages[conversationId];
          
          return {
            conversations: newConversations,
            messages: newMessages,
            currentConversationId: newConversations.length > 0 
              ? newConversations[0].id 
              : null
          };
        });
      }
    }),
    {
      name: 'ai-chat-storage',
      partialize: (state) => ({
        conversations: state.conversations,
        messages: state.messages,
        currentConversationId: state.currentConversationId,
        model: state.model,
        temperature: state.temperature,
        maxTokens: state.maxTokens
      })
    }
  )
);

export default useAIStore;
```

### 2.1.3 错误处理策略

**错误类型与处理**：

| 错误类型 | 表现 | 处理策略 | 重试机制 |
|----------|------|----------|----------|
| **网络错误** | 连接失败、超时 | 显示网络错误提示，提供重试按钮 | 指数退避重试 |
| **API错误** | 401、403、429 | 显示具体错误信息，引导用户操作 | 按错误类型决定 |
| **模型错误** | 生成失败、内容违规 | 显示友好错误提示，建议用户调整输入 | 自动重试（最多3次） |
| **前端错误** | 渲染错误、状态异常 | 显示通用错误提示，记录错误日志 | 重置状态或组件 |

**错误处理实现**：

```javascript
// errorHandler.js - 错误处理工具
class ErrorHandler {
  static handleError(error, context = {}) {
    console.error('Error in', context, error);
    
    if (error.response) {
      // API错误
      const { status, data } = error.response;
      
      switch (status) {
        case 401:
          return {
            type: 'AUTH_ERROR',
            message: '认证失败，请重新登录',
            actionable: true,
            action: 'login'
          };
        case 403:
          return {
            type: 'PERMISSION_ERROR',
            message: '权限不足，无法访问该资源',
            actionable: false
          };
        case 429:
          return {
            type: 'RATE_LIMIT_ERROR',
            message: '请求过于频繁，请稍后再试',
            actionable: true,
            action: 'wait',
            retryAfter: data.retry_after || 60
          };
        case 500:
          return {
            type: 'SERVER_ERROR',
            message: '服务器内部错误，请稍后再试',
            actionable: true,
            action: 'retry'
          };
        default:
          return {
            type: 'API_ERROR',
            message: data.error || `API错误 (${status})`,
            actionable: true,
            action: 'retry'
          };
      }
    } else if (error.request) {
      // 网络错误
      return {
        type: 'NETWORK_ERROR',
        message: '网络连接失败，请检查网络设置',
        actionable: true,
        action: 'retry'
      };
    } else {
      // 前端错误
      return {
        type: 'CLIENT_ERROR',
        message: error.message || '操作失败，请重试',
        actionable: true,
        action: 'retry'
      };
    }
  }
  
  static async retryWithBackoff(fn, maxRetries = 3, baseDelay = 1000) {
    let lastError;
    
    for (let i = 0; i < maxRetries; i++) {
      try {
        return await fn();
      } catch (error) {
        lastError = error;
        
        // 检查是否可重试
        const errorInfo = this.handleError(error);
        if (errorInfo.action !== 'retry') {
          throw error;
        }
        
        // 指数退避
        const delay = baseDelay * Math.pow(2, i) + Math.random() * 1000;
        console.log(`Retrying in ${delay}ms...`);
        await new Promise(resolve => setTimeout(resolve, delay));
      }
    }
    
    throw lastError;
  }
  
  static logError(error, context = {}) {
    // 这里可以集成错误监控服务
    console.error('Error logged:', {
      error: error.message,
      stack: error.stack,
      context,
      timestamp: new Date().toISOString()
    });
  }
}

// 使用示例
async function fetchAIResponse(messages) {
  try {
    const response = await ErrorHandler.retryWithBackoff(async () => {
      return await aiClient.chatComplete(messages);
    });
    return response;
  } catch (error) {
    const errorInfo = ErrorHandler.handleError(error, { context: 'AI response fetch' });
    ErrorHandler.logError(error, { context: 'AI response fetch' });
    throw errorInfo;
  }
}
```

```jsx
const ChatInterface = () => {
  const [input, setInput] = useState('');
  const [isStreaming, setIsStreaming] = useState(false);
  const messagesEndRef = useRef(null);
  
  const {
    currentConversationId,
    messages,
    addMessage,
    updateMessage,
    setGenerating,
    setError
  } = useAIStore();
  
  const conversationMessages = messages[currentConversationId] || [];
  
  // 滚动到最新消息
  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  }, [conversationMessages]);
  
  const handleSend = async () => {
    if (!input.trim() || isStreaming) return;
    
    const userMessage = {
      id: Date.now().toString(),
      role: 'user',
      content: input.trim(),
      timestamp: new Date().toISOString()
    };
    
    addMessage(currentConversationId, userMessage);
    setInput('');
    setIsStreaming(true);
    setGenerating(true);
    
    try {
      const assistantMessage = {
        id: (Date.now() + 1).toString(),
        role: 'assistant',
        content: '',
        timestamp: new Date().toISOString(),
        isStreaming: true
      };
      
      addMessage(currentConversationId, assistantMessage);
      
      // 启动流式生成
      await startStreaming(
        [userMessage, assistantMessage],
        (chunk) => {
          // 处理流式数据
          if (chunk.choices[0].delta.content) {
            updateMessage(
              currentConversationId,
              assistantMessage.id,
              {
                content: assistantMessage.content + chunk.choices[0].delta.content
              }
            );
          }
        },
        () => {
          // 流式结束
          updateMessage(
            currentConversationId,
            assistantMessage.id,
            { isStreaming: false }
          );
          setIsStreaming(false);
          setGenerating(false);
        },
        (error) => {
          // 错误处理
          setError(error);
          setIsStreaming(false);
          setGenerating(false);
        }
      );
    } catch (error) {
      setError(error);
      setIsStreaming(false);
      setGenerating(false);
    }
  };
  
  const handleKeyPress = (e) => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault();
      handleSend();
    }
  };
  
  return (
    <div className="chat-interface">
      {/* 消息列表 */}
      <div className="messages-container">
        {conversationMessages.map((message) => (
          <div 
            key={message.id} 
            className={`message ${message.role === 'user' ? 'user-message' : 'assistant-message'}`}
          >
            <div className="message-role">
              {message.role === 'user' ? 'You' : 'AI'}
            </div>
            <div className="message-content">
              {message.content}
              {message.isStreaming && (
                <span className="typing-indicator">
                  <span>.</span>
                  <span>.</span>
                  <span>.</span>
                </span>
              )}
            </div>
            <div className="message-time">
              {new Date(message.timestamp).toLocaleTimeString()}
            </div>
          </div>
        ))}
        <div ref={messagesEndRef} />
      </div>
      
      {/* 输入区域 */}
      <div className="input-container">
        <textarea
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyPress={handleKeyPress}
          placeholder="Type your message..."
          disabled={isStreaming}
          className="message-input"
        />
        <button
          onClick={handleSend}
          disabled={!input.trim() || isStreaming}
          className="send-button"
        >
          {isStreaming ? 'Sending...' : 'Send'}
        </button>
      </div>
    </div>
  );
};
```

```javascript
// services/api.js
import axios from 'axios';

class AIService {
  constructor() {
    this.client = axios.create({
      baseURL: import.meta.env.VITE_OPENAI_BASE_URL,
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${import.meta.env.VITE_OPENAI_API_KEY}`
      },
      timeout: 30000
    });
  }

  // 发送消息
  async sendMessage(messages, model = 'gpt-3.5-turbo') {
    try {
      const response = await this.client.post('/chat/completions', {
        model,
        messages,
        temperature: 0.7,
        max_tokens: 1000
      });
      return response.data;
    } catch (error) {
      console.error('Error sending message:', error);
      throw error;
    }
  }

  // 流式发送消息
  async streamMessage(messages, model = 'gpt-3.5-turbo', onChunk, onComplete, onError) {
    try {
      const response = await this.client.post('/chat/completions', {
        model,
        messages,
        temperature: 0.7,
        max_tokens: 1000,
        stream: true
      }, {
        responseType: 'stream'
      });

      const reader = response.data.getReader();
      const decoder = new TextDecoder();
      let fullResponse = '';

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        const chunk = decoder.decode(value, { stream: true });
        const lines = chunk.split('\n');

        for (const line of lines) {
          if (line.startsWith('data: ')) {
            const data = line.substring(6);
            if (data === '[DONE]') {
              onComplete(fullResponse);
              return;
            }
            try {
              const parsed = JSON.parse(data);
              const content = parsed.choices[0]?.delta?.content;
              if (content) {
                fullResponse += content;
                onChunk(content);
              }
            } catch (e) {
              console.error('Error parsing stream:', e);
            }
          }
        }
      }
    } catch (error) {
      console.error('Error streaming message:', error);
      onError(error);
    }
  }
}

export default new AIService();
```

```javascript
// services/errorHandler.js
class ErrorHandler {
  static handleError(error) {
    console.error('Error:', error);

    if (error.response) {
      // API错误
      const { status, data } = error.response;

      switch (status) {
        case 401:
          return {
            type: 'AUTH_ERROR',
            message: '认证失败，请检查API密钥',
            actionable: true
          };
        case 429:
          return {
            type: 'RATE_LIMIT_ERROR',
            message: '请求过于频繁，请稍后再试',
            actionable: true,
            retryAfter: data.retry_after || 60
          };
        case 500:
          return {
            type: 'SERVER_ERROR',
            message: '服务器错误，请稍后再试',
            actionable: true
          };
        default:
          return {
            type: 'API_ERROR',
            message: data.error || `API错误 (${status})`,
            actionable: false
          };
      }
    } else if (error.request) {
      // 网络错误
      return {
        type: 'NETWORK_ERROR',
        message: '网络连接失败，请检查网络设置',
        actionable: true
      };
    } else {
      // 前端错误
      return {
        type: 'CLIENT_ERROR',
        message: error.message || '操作失败，请重试',
        actionable: true
      };
    }
  }

  static async retryWithBackoff(fn, maxRetries = 3, baseDelay = 1000) {
    let lastError;

    for (let i = 0; i < maxRetries; i++) {
      try {
        return await fn();
      } catch (error) {
        lastError = error;
        
        // 检查是否可重试
        const errorInfo = this.handleError(error);
        if (!errorInfo.actionable) {
          throw error;
        }

        // 指数退避
        const delay = baseDelay * Math.pow(2, i) + Math.random() * 1000;
        await new Promise(resolve => setTimeout(resolve, delay));
      }
    }

    throw lastError;
  }
}

export default ErrorHandler;
```

### 4.2.2 Transformers.js

**优势**：
- **易用性**：简单的API，与Hugging Face模型兼容
- **性能**：优化的推理速度
- **生态**：丰富的预训练模型

**实现示例**：

```javascript
// services/transformersService.js
import { AutoModel, AutoTokenizer } from '@xenova/transformers';

class TransformersService {
  constructor() {
    this.model = null;
    this.tokenizer = null;
  }

  // 加载模型
  async loadModel(modelName = 'Xenova/distilbert-base-uncased-finetuned-sst-2-english') {
    try {
      console.log('Loading model:', modelName);
      
      // 加载分词器
      this.tokenizer = await AutoTokenizer.from_pretrained(modelName);
      
      // 加载模型
      this.model = await AutoModel.from_pretrained(modelName);
      
      console.log('Model loaded successfully');
    } catch (error) {
      console.error('Error loading model:', error);
      throw error;
    }
  }

  // 情感分析
  async sentimentAnalysis(text) {
    if (!this.model || !this.tokenizer) {
      throw new Error('Model not loaded');
    }

    try {
      // 分词
      const inputs = this.tokenizer(text, { return_tensors: 'tf' });
      
      // 推理
      const outputs = await this.model(inputs);
      
      // 处理输出
      const logits = outputs.logits;
      const probabilities = tf.softmax(logits).dataSync();
      
      // 转换为结果
      const labels = ['negative', 'positive'];
      const predictedIndex = tf.argMax(logits, 1).dataSync()[0];
      const predictedLabel = labels[predictedIndex];
      const confidence = probabilities[predictedIndex];
      
      return {
        label: predictedLabel,
        confidence: confidence
      };
    } catch (error) {
      console.error('Error during sentiment analysis:', error);
      throw error;
    }
  }
}

export default new TransformersService();
```

### 4.2.3 混合推理策略

**本地+云端混合**：
- **本地推理**：处理简单任务，保证响应速度
- **云端推理**：处理复杂任务，提供更强大的能力
- **智能切换**：根据任务类型和网络状况自动切换

**实现示例**：

```javascript
// services/hybridInference.js
import localInference from './localInference';
import aiService from './api';

class HybridInference {
  constructor() {
    this.localModels = {
      sentiment: 'models/sentiment.onnx',
      textClassification: 'models/text-classification.onnx'
    };
  }

  // 智能推理
  async infer(task, input) {
    try {
      // 检查网络状态
      const isOnline = navigator.onLine;
      
      // 根据任务类型选择推理方式
      switch (task) {
        case 'sentiment':
          // 简单任务，使用本地推理
          if (isOnline) {
            await localInference.loadModel(this.localModels.sentiment);
            return await localInference.infer(input);
          } else {
            // 离线时使用云端
            return await aiService.sendMessage([
              { role: 'user', content: `Analyze the sentiment of this text: ${input}` }
            ]);
          }
        
        case 'chat':
          // 复杂任务，使用云端推理
          return await aiService.sendMessage(input);
        
        default:
          throw new Error('Unknown task type');
      }
    } catch (error) {
      console.error('Error during hybrid inference:', error);
      //  fallback to cloud if local fails
      return await aiService.sendMessage([
        { role: 'user', content: `Process this input for task ${task}: ${input}` }
      ]);
    }
  }
}

export default new HybridInference();
```
