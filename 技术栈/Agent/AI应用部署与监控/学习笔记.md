### 5.1.2 AI应用监控的特殊性

| 监控维度 | 传统应用 | AI应用 | 关键指标 |
|----------|----------|--------|----------|
| **系统资源** | CPU、内存 | GPU显存、利用率 | GPU内存使用、温度、功耗 |
| **性能指标** | 响应时间 | Token生成速度 | TTFT、TPOT、吞吐量 |
| **业务指标** | 请求数 | 模型输出质量 | 幻觉率、相关性评分 |
| **成本指标** | 较少关注 | 推理成本 | $/1K tokens、GPU时长 |

**AI监控关键指标解释**：

- **TTFT (Time To First Token)**：首Token延迟，用户体验关键指标
- **TPOT (Time Per Output Token)**：每个输出Token的生成时间
- **Throughput**：每秒生成的Token数（tokens/s）
- **GPU利用率**：GPU计算单元使用率，低利用率表示资源浪费
- **GPU显存占用**：模型权重 + KV Cache + 激活值

```yaml
      - alert: HighLatency
```

### 5.2.2 AI服务指标暴露（Python）

```python
# metrics.py - AI服务指标暴露
from prometheus_client import Counter, Histogram, Gauge, Info, generate_latest, CONTENT_TYPE_LATEST
from fastapi import FastAPI, Request, Response
import time

# 定义指标
REQUEST_COUNT = Counter(
    'http_requests_total',
    'Total HTTP requests',
    ['method', 'endpoint', 'status']
)

REQUEST_LATENCY = Histogram(
    'http_request_duration_seconds',
    'HTTP request latency',
    ['method', 'endpoint'],
    buckets=[0.01, 0.05, 0.1, 0.25, 0.5, 1.0, 2.5, 5.0, 10.0]
)

# AI特定指标
INFERENCE_TOKENS = Counter(
    'inference_tokens_generated_total',
    'Total tokens generated',
    ['model_name']
)

INFERENCE_LATENCY = Histogram(
    'inference_duration_seconds',
    'Inference latency',
    ['model_name'],
    buckets=[0.1, 0.5, 1.0, 2.0, 5.0, 10.0, 30.0]
)

GPU_MEMORY_USED = Gauge(
    'gpu_memory_used_bytes',
    'GPU memory used',
    ['gpu_id']
)

GPU_UTILIZATION = Gauge(
    'gpu_utilization_percent',
    'GPU utilization percentage',
    ['gpu_id']
)

MODEL_INFO = Info(
    'model',
    'Model information'
)

QUEUE_SIZE = Gauge(
    'inference_queue_size',
    'Current inference queue size'
)

BATCH_SIZE = Histogram(
    'inference_batch_size',
    'Inference batch size',
    buckets=[1, 2, 4, 8, 16, 32]
)

def setup_metrics(app: FastAPI):
    """配置FastAPI指标收集"""
    
    @app.middleware("http")
    async def metrics_middleware(request: Request, call_next):
        start_time = time.time()
        
        response = await call_next(request)
        
        # 记录延迟
        duration = time.time() - start_time
        REQUEST_LATENCY.labels(
            method=request.method,
            endpoint=request.url.path
        ).observe(duration)
        
        # 记录请求数
        REQUEST_COUNT.labels(
            method=request.method,
            endpoint=request.url.path,
            status=response.status_code
        ).inc()
        
        return response
    
    @app.get("/metrics")
    async def metrics():
        """Prometheus指标端点"""
        return Response(
            content=generate_latest(),
            media_type=CONTENT_TYPE_LATEST
        )

# 在推理代码中使用
class InferenceMetrics:
    """推理指标收集器"""
    
    @staticmethod
    def record_inference(model_name: str, tokens_generated: int, duration: float):
        """记录推理指标"""
        INFERENCE_TOKENS.labels(model_name=model_name).inc(tokens_generated)
        INFERENCE_LATENCY.labels(model_name=model_name).observe(duration)
    
    @staticmethod
    def update_gpu_metrics():
        """更新GPU指标（需要nvidia-ml-py）"""
        try:
            import pynvml
            pynvml.nvmlInit()
            
            device_count = pynvml.nvmlDeviceGetCount()
            for i in range(device_count):
                handle = pynvml.nvmlDeviceGetHandleByIndex(i)
                
                # 显存使用
                mem_info = pynvml.nvmlDeviceGetMemoryInfo(handle)
                GPU_MEMORY_USED.labels(gpu_id=str(i)).set(mem_info.used)
                
                # GPU利用率
                util = pynvml.nvmlDeviceGetUtilizationRates(handle)
                GPU_UTILIZATION.labels(gpu_id=str(i)).set(util.gpu)
                
        except Exception as e:
            print(f"Failed to get GPU metrics: {e}")
    
    @staticmethod
    def set_model_info(model_name: str, version: str, parameters: str):
        """设置模型信息"""
        MODEL_INFO.info({
            'name': model_name,
            'version': version,
            'parameters': parameters
        })
```

## 5.4 链路追踪

### 5.4.1 OpenTelemetry + Jaeger

```python
# tracing.py - 分布式链路追踪
from opentelemetry import trace
from opentelemetry.exporter.jaeger.thrift import JaegerExporter
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import BatchSpanProcessor
from opentelemetry.instrumentation.fastapi import FastAPIInstrumentor
from opentelemetry.instrumentation.requests import RequestsInstrumentor
from functools import wraps
import time

# 配置Tracer
provider = TracerProvider()
processor = BatchSpanProcessor(
    JaegerExporter(
        agent_host_name="jaeger-agent",
        agent_port=6831,
    )
)
provider.add_span_processor(processor)
trace.set_tracer_provider(provider)

tracer = trace.get_tracer(__name__)

def setup_tracing(app):
    """配置FastAPI链路追踪"""
    FastAPIInstrumentor.instrument_app(app)
    RequestsInstrumentor().instrument()

def trace_span(name: str, attributes: dict = None):
    """装饰器：为函数添加链路追踪"""
    def decorator(func):
        @wraps(func)
        async def async_wrapper(*args, **kwargs):
            with tracer.start_as_current_span(name) as span:
                if attributes:
                    for key, value in attributes.items():
                        span.set_attribute(key, value)
                
                start_time = time.time()
                try:
                    result = await func(*args, **kwargs)
                    span.set_attribute("success", True)
                    return result
                except Exception as e:
                    span.set_attribute("error", True)
                    span.set_attribute("error.message", str(e))
                    raise
                finally:
                    duration = time.time() - start_time
                    span.set_attribute("duration_ms", duration * 1000)
        
        @wraps(func)
        def sync_wrapper(*args, **kwargs):
            with tracer.start_as_current_span(name) as span:
                if attributes:
                    for key, value in attributes.items():
                        span.set_attribute(key, value)
                
                start_time = time.time()
                try:
                    result = func(*args, **kwargs)
                    span.set_attribute("success", True)
                    return result
                except Exception as e:
                    span.set_attribute("error", True)
                    span.set_attribute("error.message", str(e))
                    raise
                finally:
                    duration = time.time() - start_time
                    span.set_attribute("duration_ms", duration * 1000)
        
        return async_wrapper if asyncio.iscoroutinefunction(func) else sync_wrapper
    return decorator

# 在AI服务中使用
class InferenceTracer:
    """推理链路追踪"""
    
    @staticmethod
    @trace_span("model_inference", {"component": "model"})
    async def trace_inference(model_name: str, prompt: str, max_tokens: int):
        """追踪模型推理"""
        with tracer.start_as_current_span("tokenization") as span:
            span.set_attribute("prompt_length", len(prompt))
            # Tokenize...
        
        with tracer.start_as_current_span("generation") as span:
            span.set_attribute("max_tokens", max_tokens)
            # Generate...
        
        with tracer.start_as_current_span("postprocessing") as span:
            # Postprocess...
            pass
```

## 5.5 本章小结

| 知识点 | 面试关键词 | 实际应用 |
|--------|------------|----------|
| 可观测性 | Metrics/Logs/Traces | 构建完整监控体系 |
| Prometheus | 指标采集、告警规则、TSDB | AI服务性能监控 |
| Grafana | 仪表盘、可视化、告警 | 实时监控大屏 |
| 日志聚合 | Loki、Promtail、结构化日志 | 日志查询与分析 |
| 链路追踪 | OpenTelemetry、Jaeger、Span | 请求链路分析 |

### 常见坑与注意点

| 现象 | 原因 | 正确做法 |
|------|------|----------|
| 指标数据缺失 | 抓取间隔过长或服务未暴露/metrics | 设置合理的scrape_interval，确保服务暴露指标端点 |
| 告警风暴 | 告警阈值设置不当，无抑制规则 | 使用Alertmanager分组和抑制，设置合理的for时间 |
| 日志查询慢 | 日志量过大，未建立索引 | 使用结构化日志，配置合理的保留策略 |

### 面试常见问题

**Q1: Prometheus的抓取机制是什么？**

> **回答要点**：
> 1. **Pull模式**：Prometheus主动拉取目标指标
> 2. **服务发现**：支持静态配置和动态发现（K8s、Consul等）
> 3. **抓取流程**：按配置间隔HTTP GET /metrics，解析文本格式数据
> 4. **优势**：易于排查问题，目标服务无需推送逻辑

**Q2: AI服务需要监控哪些特殊指标？**

> **回答要点**：
> 1. **推理性能**：TTFT（首Token延迟）、TPOT（每Token时间）、吞吐量

> 3. **模型质量**：幻觉率、相关性评分、用户反馈
> 4. **成本指标**：$/1K tokens、GPU使用时长

**Q3: 如何实现分布式链路追踪？**

> **回答要点**：
> 1. **Trace**：完整的请求链路
> 2. **Span**：链路中的单个操作，包含开始时间、持续时间、标签
> 3. **Context传播**：通过HTTP Header传递Trace ID
> 4. **采样策略**：概率采样或基于延迟的采样
> 5. **工具**：OpenTelemetry + Jaeger/Zipkin

```python
# batch_inference.py - 动态批处理优化
import asyncio
from typing import List, Dict
import time
from dataclasses import dataclass

@dataclass
class InferenceRequest:
    id: str
    prompt: str
    max_tokens: int
    future: asyncio.Future
    timestamp: float

class DynamicBatcher:
    """动态批处理器 - 提高GPU利用率"""
    
    def __init__(self, model, max_batch_size=8, max_wait_ms=50):
        self.model = model
        self.max_batch_size = max_batch_size
        self.max_wait_ms = max_wait_ms / 1000  # 转换为秒
        
        self.queue: List[InferenceRequest] = []
        self.lock = asyncio.Lock()
        self.processing = False
    
    async def submit(self, prompt: str, max_tokens: int) -> str:
        """提交推理请求"""
        future = asyncio.get_event_loop().create_future()
        request = InferenceRequest(
            id=str(time.time()),
            prompt=prompt,
            max_tokens=max_tokens,
            future=future,
            timestamp=time.time()
        )
        
        async with self.lock:
            self.queue.append(request)
        
        # 触发批处理
        if not self.processing:
            asyncio.create_task(self._process_batch())
        
        return await future
    
    async def _process_batch(self):
        """处理批次"""
        self.processing = True
        
        while True:
            async with self.lock:
                if not self.queue:
                    self.processing = False
                    return
                
                # 收集批次
                batch = []
                wait_start = time.time()
                
                while len(batch) < self.max_batch_size:
                    if self.queue:
                        batch.append(self.queue.pop(0))
                    
                    # 等待更多请求或达到最大等待时间
                    if len(batch) >= 2 or (time.time() - wait_start) > self.max_wait_ms:
                        break
                    
                    await asyncio.sleep(0.01)
            
            if batch:
                await self._execute_batch(batch)
```

```python
# model_storage_optimizer.py - 模型存储优化
import os
import shutil
from pathlib import Path
import hashlib
import json

class ModelStorageOptimizer:
    """模型存储优化器"""
    
    def __init__(self, cache_dir: str = "/models"):
        self.cache_dir = Path(cache_dir)
        self.cache_dir.mkdir(parents=True, exist_ok=True)
        self.metadata_file = self.cache_dir / "cache_metadata.json"
        self.metadata = self._load_metadata()
    
    def _load_metadata(self) -> dict:
        """加载缓存元数据"""
        if self.metadata_file.exists():
            with open(self.metadata_file) as f:
                return json.load(f)
        return {}
    
    def _save_metadata(self):
        """保存缓存元数据"""
        with open(self.metadata_file, 'w') as f:
            json.dump(self.metadata, f, indent=2)
    
    def get_model_path(self, model_name: str, revision: str = "main") -> Path:
        """获取模型路径（带缓存）"""
        cache_key = f"{model_name}@{revision}"
        
        if cache_key in self.metadata:
            model_path = self.cache_dir / self.metadata[cache_key]["local_path"]
            if model_path.exists():
                # 更新访问时间
                self.metadata[cache_key]["last_accessed"] = time.time()
                self._save_metadata()
                return model_path
        
        # 下载模型
        return self._download_model(model_name, revision)
    
    def _download_model(self, model_name: str, revision: str) -> Path:
        """下载并缓存模型"""
        from huggingface_hub import snapshot_download
        
        local_path = snapshot_download(
            model_name,
            revision=revision,
            cache_dir=str(self.cache_dir),
            local_dir_use_symlinks=False
        )
        
        # 记录元数据
        cache_key = f"{model_name}@{revision}"
        self.metadata[cache_key] = {
            "local_path": local_path,
            "downloaded_at": time.time(),
            "last_accessed": time.time(),
            "size_mb": self._get_dir_size(local_path)
        }
        self._save_metadata()
        
        return Path(local_path)
    
    def cleanup_old_models(self, max_age_days: int = 30, max_cache_gb: float = 100):
        """清理旧模型"""
        current_time = time.time()
        max_age_seconds = max_age_days * 24 * 3600
        
        # 按访问时间排序
        sorted_models = sorted(
            self.metadata.items(),
            key=lambda x: x[1]["last_accessed"]
        )
        
        total_size = sum(m["size_mb"] for _, m in sorted_models)
        max_size_mb = max_cache_gb * 1024
        
        removed = []
        for cache_key, meta in sorted_models:
            age = current_time - meta["last_accessed"]
            
            # 删除条件：超过最大年龄或缓存过大
            if age > max_age_seconds or total_size > max_size_mb:
                model_path = self.cache_dir / meta["local_path"]
                if model_path.exists():
                    shutil.rmtree(model_path)
                    removed.append(cache_key)
                    total_size -= meta["size_mb"]
        
        # 更新元数据
        for key in removed:
            del self.metadata[key]
        self._save_metadata()
        
        print(f"Cleaned up {len(removed)} models, freed {sum(self.metadata[k]['size_mb'] for k in removed):.2f} MB")
    
    def quantize_and_cache(self, model_name: str, quantization: str = "int8"):
        """量化模型并缓存"""
        from transformers import AutoModelForCausalLM, BitsAndBytesConfig
        
        cache_key = f"{model_name}@{quantization}"
        quantized_path = self.cache_dir / f"quantized_{cache_key.replace('/', '_')}"
        
        if quantized_path.exists():
            return quantized_path
        
        # 量化配置
        if quantization == "int8":
            config = BitsAndBytesConfig(load_in_8bit=True)
        elif quantization == "int4":
            config = BitsAndBytesConfig(load_in_4bit=True)
        else:
            raise ValueError(f"Unsupported quantization: {quantization}")
        
        # 加载并量化
        model = AutoModelForCausalLM.from_pretrained(
            model_name,
            quantization_config=config,
            device_map="auto"
        )
        
        # 保存量化模型
        model.save_pretrained(quantized_path)
        
        return quantized_path
    
    @staticmethod
    def _get_dir_size(path: str) -> float:
        """获取目录大小（MB）"""
        total = 0
        for dirpath, dirnames, filenames in os.walk(path):
            for f in filenames:
                fp = os.path.join(dirpath, f)
                total += os.path.getsize(fp)
        return total / (1024 * 1024)
```

### 面试常见问题

**Q1: 如何降低AI推理服务的成本？**

> **回答要点**：
> 1. **优势**：提高GPU利用率、增加吞吐量、降低单请求成本
> 2. **实现**：设置最大batch size和最大等待时间
> 3. **注意**：平衡延迟和吞吐量，避免等待时间过长
> 4. **适用**：对延迟不敏感的离线场景效果最佳
