## 🚀 快速开始

### 连接MongoDB

```bash

mongosh

# 指定主机端口
mongosh "mongodb://localhost:27017"

# 认证连接

# 连接字符串（复制集）
mongosh "mongodb://host1:27017,host2:27018,host3:27019/mydb?replicaSet=rs0"

# 连接字符串（分片集群）
mongosh "mongodb://mongos1:27017,mongos2:27017/mydb"
```

## 📝 文档CRUD操作

### 插入文档

```javascript
// 插入单条
db.users.insertOne({
  name: "张三",
  age: 25,
  email: "zhangsan@example.com"
})

// 插入多条
db.users.insertMany([
  { name: "李四", age: 30 },
  { name: "王五", age: 28 }
])

// 批量插入（大批量时使用）
db.users.insertMany([...], { ordered: false })  // 遇错继续
```

### 查询文档

```javascript
// 查询所有
db.users.find()

// 条件查询
db.users.find({ age: 25 })
db.users.find({ age: { $gte: 20, $lte: 30 } })
db.users.find({ name: /^张/ })

// 投影
db.users.find({}, { name: 1, age: 1, _id: 0 })

// 排序
db.users.find().sort({ age: 1 })   // 升序
db.users.find().sort({ age: -1 })  // 降序

// 分页
db.users.find().skip(0).limit(10)
db.users.find().skip(10).limit(10)

// 查询单条
db.users.findOne({ name: "张三" })

// 统计数量
db.users.countDocuments({ age: { $gte: 25 } })

// 去重
db.users.distinct("city")
```

### 更新文档

```javascript
// 更新单条
db.users.updateOne(
  { name: "张三" },
  { $set: { age: 26 } }
)

// 更新多条
db.users.updateMany(
  { age: { $lt: 30 } },
  { $set: { status: "young" } }
)

// 替换文档
db.users.replaceOne(
  { name: "张三" },
  { name: "张三", age: 26, email: "new@example.com" }
)

// Upsert（不存在则插入）
db.users.updateOne(
  { name: "张三" },
  { $set: { age: 26 } },
  { upsert: true }
)

// 数组操作
db.users.updateOne({ name: "张三" }, { $push: { tags: "java" } })
db.users.updateOne({ name: "张三" }, { $pull: { tags: "java" } })
db.users.updateOne({ name: "张三" }, { $addToSet: { tags: "java" } })  // 去重添加

// 字段操作
db.users.updateOne({ name: "张三" }, { $inc: { age: 1 } })      // 自增
db.users.updateOne({ name: "张三" }, { $unset: { temp: "" } })  // 删除字段
db.users.updateOne({ name: "张三" }, { $rename: { "old": "new" } })  // 重命名
```

### 删除文档

```javascript
// 删除单条
db.users.deleteOne({ name: "张三" })

// 删除多条
db.users.deleteMany({ age: { $lt: 25 } })

// 删除所有
db.users.deleteMany({})
```

## 🔍 查询操作符速查

### 比较操作符

| 操作符 | 含义 | 示例 |
|--------|------|------|
| `$eq` | 等于 | `{ age: { $eq: 25 } }` |
| `$ne` | 不等于 | `{ age: { $ne: 25 } }` |
| `$gt` | 大于 | `{ age: { $gt: 25 } }` |
| `$gte` | 大于等于 | `{ age: { $gte: 25 } }` |
| `$lt` | 小于 | `{ age: { $lt: 25 } }` |
| `$lte` | 小于等于 | `{ age: { $lte: 25 } }` |
| `$in` | 在数组中 | `{ age: { $in: [20, 25, 30] } }` |
| `$nin` | 不在数组中 | `{ age: { $nin: [20, 25] } }` |

### 逻辑操作符

```javascript
// AND（逗号分隔）
db.users.find({ age: 25, name: "张三" })

// OR
db.users.find({ $or: [{ age: 25 }, { age: 30 }] })

// NOT
db.users.find({ age: { $not: { $gt: 25 } } })

// NOR（都不匹配）
db.users.find({ $nor: [{ age: 25 }, { age: 30 }] })
```

### 数组操作符

```javascript
// 包含元素
db.users.find({ tags: "java" })

// 包含所有元素
db.users.find({ tags: { $all: ["java", "mongodb"] } })

// 数组大小
db.users.find({ tags: { $size: 3 } })

// 数组元素匹配
db.users.find({ scores: { $elemMatch: { $gte: 80, $lt: 90 } } })
```

### 元素操作符

```javascript
// 字段存在
db.users.find({ email: { $exists: true } })

// 字段类型
db.users.find({ age: { $type: "int" } })
```

## 📊 聚合管道速查

### 常用阶段

```javascript
// $match - 过滤
{ $match: { status: "active" } }

// $group - 分组
{ $group: { _id: "$city", count: { $sum: 1 }, avgAge: { $avg: "$age" } } }

// $sort - 排序
{ $sort: { count: -1 } }

// $project - 投影
{ $project: { name: 1, age: 1, _id: 0 } }

// $limit - 限制
{ $limit: 10 }

// $skip - 跳过
{ $skip: 10 }

// $unwind - 展开数组
{ $unwind: "$tags" }

// $lookup - 关联查询
{
  $lookup: {
    from: "orders",
    localField: "_id",
    foreignField: "userId",
    as: "orders"
  }
}

// $addFields - 添加字段
{ $addFields: { fullName: { $concat: ["$firstName", " ", "$lastName"] } } }

// $facet - 多面聚合
{
  $facet: {
    totalCount: [{ $count: "count" }],
    byStatus: [{ $group: { _id: "$status", count: { $sum: 1 } } }]
  }
}
```

### 聚合累加器

| 累加器 | 说明 | 示例 |
|--------|------|------|
| `$sum` | 求和 | `{ $sum: 1 }` 或 `{ $sum: "$amount" }` |
| `$avg` | 平均值 | `{ $avg: "$age" }` |
| `$min` | 最小值 | `{ $min: "$price" }` |
| `$max` | 最大值 | `{ $max: "$price" }` |
| `$push` | 收集为数组 | `{ $push: "$name" }` |
| `$addToSet` | 去重收集 | `{ $addToSet: "$tag" }` |
| `$first` | 第一个 | `{ $first: "$name" }` |
| `$last` | 最后一个 | `{ $last: "$name" }` |

### 聚合示例

```javascript
// 统计各城市用户数
db.users.aggregate([
  { $group: { _id: "$city", count: { $sum: 1 } } },
  { $sort: { count: -1 } }
])

// 关联查询订单
db.users.aggregate([
  { $match: { status: "active" } },
  {
    $lookup: {
      from: "orders",
      localField: "_id",
      foreignField: "userId",
      as: "userOrders"
    }
  },
  { $unwind: "$userOrders" },
  {
    $group: {
      _id: "$_id",
      name: { $first: "$name" },
      totalSpent: { $sum: "$userOrders.amount" }
    }
  }
])

// 时间分组统计
db.orders.aggregate([
  {
    $group: {
      _id: {
        year: { $year: "$createTime" },
        month: { $month: "$createTime" }
      },
      totalAmount: { $sum: "$amount" },
      orderCount: { $sum: 1 }
    }
  },
  { $sort: { "_id.year": -1, "_id.month": -1 } }
])
```

## 🔧 索引操作

### 索引管理

```javascript
// 查看索引
db.users.getIndexes()

// 创建索引
db.users.createIndex({ name: 1 })                    // 单字段
db.users.createIndex({ name: 1, age: -1 })           // 复合
db.users.createIndex({ tags: 1 })                    // 多键
db.users.createIndex({ content: "text" })            // 文本
db.users.createIndex({ location: "2dsphere" })       // 地理空间
db.users.createIndex({ userId: "hashed" })           // 哈希

// 创建唯一索引
db.users.createIndex({ email: 1 }, { unique: true })

// 创建TTL索引
db.sessions.createIndex({ createdAt: 1 }, { expireAfterSeconds: 3600 })

// 删除索引
db.users.dropIndex("name_1")
db.users.dropIndex({ name: 1 })

// 查看索引大小
db.users.totalIndexSize()

// 重建索引
db.users.reIndex()
```

### 执行计划分析

```javascript
// 查看执行计划
db.users.find({ name: "张三" }).explain()

// 详细执行统计
db.users.find({ name: "张三" }).explain("executionStats")

// 关键指标
// - stage: COLLSCAN(全表扫描) / IXSCAN(索引扫描) / FETCH
// - totalDocsExamined: 扫描文档数
// - totalKeysExamined: 扫描索引键数
// - nReturned: 返回文档数
// - executionTimeMillis: 执行时间
```

## 🖥️ 复制集操作

### 复制集管理

```javascript
// 初始化复制集
rs.initiate({
  _id: "rs0",
  members: [
    { _id: 0, host: "localhost:27017" },
    { _id: 1, host: "localhost:27018" },
    { _id: 2, host: "localhost:27019" }
  ]
})

// 查看状态
rs.status()

// 查看配置
rs.conf()

// 添加成员
rs.add("localhost:27020")
rs.addArb("localhost:27021")  // 仲裁节点

// 移除成员
rs.remove("localhost:27020")

// 重新配置
cfg = rs.conf()
cfg.members[0].priority = 2
rs.reconfig(cfg)

// 主节点降级
rs.stepDown()

// 查看复制延迟
rs.printSecondaryReplicationInfo()

// 查看主节点
rs.isMaster()
```

### 读写分离配置

```javascript
// 连接字符串配置
mongodb://host1:27017,host2:27018/mydb?replicaSet=rs0&readPreference=secondary

// 读偏好选项
// primary           - 只从主节点读
// primaryPreferred  - 优先主节点
// secondary         - 只从从节点读
// secondaryPreferred - 优先从节点
// nearest           - 延迟最低的节点

// 写关注配置
mongodb://host1:27017/mydb?w=majority&wtimeoutMS=5000

// w=0        不等待确认
// w=1        等待主节点确认（默认）
// w=majority 等待大多数节点确认
```

### 分片管理


## 🎯 常用代码场景

### 1. 分页查询优化

```javascript
// 传统分页（skip性能差）
db.users.find().skip(10000).limit(10)

// 优化：使用范围查询
// 第一页
db.users.find().sort({ _id: 1 }).limit(10)

// 后续页（记住上一页最后一条的_id）
db.users.find({ _id: { $gt: lastId } }).sort({ _id: 1 }).limit(10)
```

### 2. 批量更新

```javascript
// 批量更新（bulkWrite）
db.users.bulkWrite([
  { updateOne: { filter: { _id: 1 }, update: { $set: { status: "active" } } } },
  { updateOne: { filter: { _id: 2 }, update: { $set: { status: "active" } } } },
  { deleteOne: { filter: { _id: 3 } } },
  { insertOne: { document: { _id: 4, name: "新用户" } } }
], { ordered: false })
```

### 3. 事务操作

```javascript
const session = db.getMongo().startSession()
session.startTransaction()

try {
  const users = session.getDatabase("mydb").users
  const orders = session.getDatabase("mydb").orders
  
  users.updateOne({ _id: 1 }, { $inc: { balance: -100 } }, { session })
  orders.insertOne({ userId: 1, amount: 100 }, { session })
  
  session.commitTransaction()
} catch (e) {
  session.abortTransaction()
} finally {
  session.endSession()
}
```

## ⚠️ 常见坑点速查

| 坑点 | 说明 | 正确做法 |
|------|------|----------|
| 深度分页 | skip越大性能越差 | 使用范围查询替代 |
| 数组无限增长 | 数组元素过多导致文档过大 | 限制数组大小或单独集合 |
| 索引未命中 | 查询条件不符合最左前缀 | 检查explain，优化索引 |
| 复制延迟 | 从节点数据落后主节点 | 监控延迟，优化硬件 |
| 分片热点 | 片键选择不当导致数据倾斜 | 选择高基数、均匀分布的片键 |
| Oplog满 | 写入量大导致Oplog覆盖 | 增大Oplog大小 |
| 连接数过多 | 连接池配置不当 | 合理配置连接池大小 |
| 内存不足 | 工作集超过内存 | 增加内存或分片 |
