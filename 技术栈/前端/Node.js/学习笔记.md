## 第1章：Node.js 概述与环境

### 1.1 Node.js 简介

- **Node.js** 是基于 Chrome V8 引擎的 JavaScript 运行时，允许在服务器端运行 JavaScript。
- **特点**：非阻塞 I/O、事件驱动、单线程模型，适合高并发场景。
- **应用场景**：后端 API、实时通信、微服务、命令行工具、构建工具等。

### 1.2 环境搭建与配置

1. **安装**：官网下载安装包或使用 nvm 管理版本
2. **版本管理**：nvm（Node Version Manager）
3. **包管理**：npm、yarn、pnpm
4. **项目初始化**：`npm init` 创建 package.json
5. **环境变量**：.env 文件与 process.env

### 1.3 模块系统

- **CommonJS**：使用 `require()` 和 `module.exports`
- **ES Module**：使用 `import` 和 `export`（需要在 package.json 中设置 "type": "module"）
- **核心模块**：fs、path、http、crypto、os 等
- **第三方模块**：通过 npm 安装和使用

## 第3章：Express 框架与 Web 开发

### 3.1 Express 简介

- **Express**：基于 Node.js 的轻量级 Web 应用框架，提供了简洁的 API 来构建 Web 服务器
- **特点**：简洁、灵活、高性能、丰富的中间件生态
- **应用场景**：构建 RESTful API、Web 应用、后端服务等

### 3.2 环境搭建与项目初始化

#### 安装 Express
```bash
npm install express
```

#### 基本示例
```javascript
const express = require('express');
const app = express();
const port = 3000;

app.get('/', (req, res) => {
  res.send('Hello World!');
});

app.listen(port, () => {
  console.log(`Server running at http://localhost:${port}`);
});
```

### 3.3 路由

#### 基本路由
```javascript
// GET 请求
app.get('/api/users', (req, res) => {
  res.json({ users: [] });
});

// POST 请求
app.post('/api/users', (req, res) => {
  const newUser = req.body;
  // 处理用户创建逻辑
  res.status(201).json(newUser);
});

// PUT 请求
app.put('/api/users/:id', (req, res) => {
  const { id } = req.params;
  const updatedUser = req.body;
  // 处理用户更新逻辑
  res.json(updatedUser);
});

// DELETE 请求
app.delete('/api/users/:id', (req, res) => {
  const { id } = req.params;
  // 处理用户删除逻辑
  res.status(204).send();
});
```

#### 路由参数
```javascript
app.get('/api/users/:id', (req, res) => {
  const { id } = req.params;
  // 根据 ID 获取用户
  res.json({ id, name: 'User' });
});
```

#### 查询参数
```javascript
app.get('/api/users', (req, res) => {
  const { page, limit } = req.query;
  // 根据分页参数获取用户列表
  res.json({ users: [], page, limit });
});
```

### 3.4 中间件

#### 内置中间件
```javascript
// 解析 JSON 请求体
app.use(express.json());

// 解析 URL 编码的请求体
app.use(express.urlencoded({ extended: true }));

// 静态文件服务
app.use(express.static('public'));
```

#### 自定义中间件
```javascript
// 日志中间件
const logger = (req, res, next) => {
  console.log(`${new Date().toISOString()} - ${req.method} ${req.url}`);
  next();
};

app.use(logger);

// 认证中间件
const auth = (req, res, next) => {
  const token = req.headers.authorization;
  if (!token) {
    return res.status(401).json({ message: 'Unauthorized' });
  }
  // 验证 token
  next();
};

app.use('/api', auth);
```

#### 错误处理中间件
```javascript
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).json({ message: 'Internal Server Error' });
});
```

## 第4章：Node.js 异步编程与高级特性

### 4.1 异步编程模型

#### 回调函数
```javascript
fs.readFile('file.txt', 'utf8', (err, data) => {
  if (err) throw err;
  console.log(data);
});
```

#### Promise
```javascript
const fs = require('fs').promises;

fs.readFile('file.txt', 'utf8')
  .then(data => {
    console.log(data);
  })
  .catch(err => {
    console.error(err);
  });
```

#### async/await
```javascript
const fs = require('fs').promises;

async function readFile() {
  try {
    const data = await fs.readFile('file.txt', 'utf8');
    console.log(data);
  } catch (err) {
    console.error(err);
  }
}

readFile();
```

### 4.2 事件循环

#### 事件循环阶段
1. **Timers**：执行 setTimeout 和 setInterval 的回调
2. **Pending Callbacks**：执行 I/O 回调
3. **Idle, Prepare**：内部使用
4. **Poll**：获取新的 I/O 事件
5. **Check**：执行 setImmediate 的回调
6. **Close Callbacks**：执行关闭事件的回调

#### 宏任务与微任务
- **宏任务**：setTimeout、setInterval、setImmediate、I/O 操作
- **微任务**：Promise.then、process.nextTick

#### 执行顺序
1. 执行同步代码
2. 执行所有微任务
3. 进入事件循环，执行宏任务
4. 执行所有微任务
5. 重复步骤 3-4

### 4.3 流（Stream）

#### 流的类型
- **Readable**：可读流
- **Writable**：可写流
- **Duplex**：双向流
- **Transform**：转换流

#### 可读流示例
```javascript
const fs = require('fs');

const readStream = fs.createReadStream('largeFile.txt');

readStream.on('data', (chunk) => {
  console.log(`Received ${chunk.length} bytes of data`);
});

readStream.on('end', () => {
  console.log('End of file');
});

readStream.on('error', (err) => {
  console.error(err);
});
```

#### 可写流示例
```javascript
const fs = require('fs');

const writeStream = fs.createWriteStream('output.txt');

writeStream.write('Hello ');
writeStream.write('World!');
writeStream.end();

writeStream.on('finish', () => {
  console.log('File written');
});
```

#### 管道操作
```javascript
const fs = require('fs');

const readStream = fs.createReadStream('input.txt');
const writeStream = fs.createWriteStream('output.txt');

readStream.pipe(writeStream);

writeStream.on('finish', () => {
  console.log('File copied');
});
```

### 4.4 子进程（Child Process）

#### 创建子进程
```javascript
const { spawn, exec, execFile, fork } = require('child_process');

// spawn 示例
const ls = spawn('ls', ['-la']);

ls.stdout.on('data', (data) => {
  console.log(`stdout: ${data}`);
});

ls.stderr.on('data', (data) => {
  console.error(`stderr: ${data}`);
});

ls.on('close', (code) => {
  console.log(`child process exited with code ${code}`);
});

// exec 示例
exec('ls -la', (error, stdout, stderr) => {
  if (error) {
    console.error(`error: ${error.message}`);
    return;
  }
  if (stderr) {
    console.error(`stderr: ${stderr}`);
    return;
  }
  console.log(`stdout: ${stdout}`);
});
```

#### 进程间通信
```javascript
const { fork } = require('child_process');

const child = fork('child.js');

// 发送消息给子进程
child.send({ message: 'Hello from parent' });

// 接收子进程的消息
child.on('message', (message) => {
  console.log(`Parent received: ${message}`);
});

// child.js
process.on('message', (message) => {
  console.log(`Child received: ${message}`);
  process.send('Hello from child');
});
```

### 4.5 集群（Cluster）

#### 集群示例
```javascript
const cluster = require('cluster');
const os = require('os');
const express = require('express');

if (cluster.isMaster) {
  console.log(`Master ${process.pid} is running`);
  
  // 创建与 CPU 核心数相同的工作进程
  for (let i = 0; i < os.cpus().length; i++) {
    cluster.fork();
  }
  
  cluster.on('exit', (worker, code, signal) => {
    console.log(`Worker ${worker.process.pid} died`);
    // 重启工作进程
    cluster.fork();
  });
} else {
  const app = express();
  
  app.get('/', (req, res) => {
    res.send(`Hello from worker ${process.pid}`);
  });
  
  app.listen(3000, () => {
    console.log(`Worker ${process.pid} is listening on port 3000`);
  });
  
  console.log(`Worker ${process.pid} started`);
}
```

## 第5章：Node.js 实战技巧

### 5.1 常见问题与解决方案

#### 回调地狱
- **问题**：多层嵌套的回调函数导致代码难以维护
- **解决方案**：使用 Promise 或 async/await

```javascript
// 回调地狱
fs.readFile('file1.txt', 'utf8', (err1, data1) => {
  if (err1) throw err1;
  fs.readFile('file2.txt', 'utf8', (err2, data2) => {
    if (err2) throw err2;
    fs.writeFile('output.txt', data1 + data2, (err3) => {
      if (err3) throw err3;
      console.log('Files written');
    });
  });
});

// 使用 Promise
const fs = require('fs').promises;

fs.readFile('file1.txt', 'utf8')
  .then(data1 => fs.readFile('file2.txt', 'utf8').then(data2 => ({ data1, data2 })))
  .then(({ data1, data2 }) => fs.writeFile('output.txt', data1 + data2))
  .then(() => console.log('Files written'))
  .catch(err => console.error(err));

// 使用 async/await
async function writeFiles() {
  try {
    const data1 = await fs.readFile('file1.txt', 'utf8');
    const data2 = await fs.readFile('file2.txt', 'utf8');
    await fs.writeFile('output.txt', data1 + data2);
    console.log('Files written');
  } catch (err) {
    console.error(err);
  }
}

writeFiles();
```

#### 内存泄漏
- **问题**：Node.js 应用内存使用持续增长
- **原因**：未释放的资源、闭包引用、事件监听器未移除
- **解决方案**：
  - 及时释放资源（数据库连接、文件流等）
  - 避免全局变量存储大量数据
  - 移除不再需要的事件监听器
  - 使用内存分析工具（如 heapdump）

#### 错误处理
- **问题**：未捕获的错误导致应用崩溃
- **解决方案**：
  - 使用 try/catch 捕获同步错误
  - 使用 .catch() 捕获 Promise 错误
  - 监听 uncaughtException 和 unhandledRejection 事件

```javascript
// 捕获未处理的 Promise 拒绝
process.on('unhandledRejection', (reason, promise) => {
  console.error('Unhandled Rejection:', reason);
  // 可以在这里记录错误、发送通知等
});

// 捕获未捕获的异常
process.on('uncaughtException', (err) => {
  console.error('Uncaught Exception:', err);
  // 可以在这里记录错误、发送通知等
  // 注意：应用可能处于不稳定状态，建议重启
  process.exit(1);
});
```
