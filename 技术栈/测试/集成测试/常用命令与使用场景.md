## Maven 常用命令

### 基本命令

```bash
# 编译项目
mvn compile

# 运行单元测试
mvn test

# 运行集成测试
mvn verify

# 运行所有测试
mvn test verify

# 跳过测试
mvn install -DskipTests

# 只运行单元测试，跳过集成测试
mvn test -DskipITs=true

# 只运行集成测试，跳过单元测试
mvn verify -DskipUnitTests=true

# 打包项目
mvn package

# 清理项目
mvn clean

# 完整构建
mvn clean install
```

### 测试相关命令

```bash
# 运行指定测试类
mvn test -Dtest=UserServiceTest

# 运行指定测试方法
mvn test -Dtest=UserServiceTest#testCreateUser

# 运行指定包下的测试
mvn test -Dtest=com.example.service.*

# 运行匹配模式的测试
mvn test -Dtest=*IntegrationTest

# 并行运行测试
mvn test -Dparallel=methods -DthreadCount=4

# 生成测试报告
mvn surefire-report:report

# 生成覆盖率报告
mvn jacoco:report

# 查看测试结果
mvn surefire:test
```

### 命令参数说明

| 参数 | 说明 |
|------|------|
| `-Dtest` | 指定测试类或方法 |
| `-DskipTests` | 跳过所有测试 |
| `-DskipITs` | 跳过集成测试 |
| `-DskipUnitTests` | 跳过单元测试 |
| `-Dparallel` | 并行执行模式 |
| `-DthreadCount` | 线程数 |
| `-DfailIfNoTests` | 没有测试时是否失败 |
| `-Dmaven.test.failure.ignore` | 忽略测试失败 |

### 常用配置

**pom.xml**

```xml
<build>
    <plugins>
        <!-- Maven Surefire Plugin (单元测试) -->
        <plugin>
            <groupId>org.apache.maven.plugins</groupId>
            <artifactId>maven-surefire-plugin</artifactId>
            <version>3.0.0-M5</version>
            <configuration>
                <parallel>methods</parallel>
                <threadCount>4</threadCount>
                <perCoreThreadCount>true</perCoreThreadCount>
                <includes>
                    <include>**/*Test.java</include>
                </includes>
                <excludes>
                    <exclude>**/*IntegrationTest.java</exclude>
                </excludes>
            </configuration>
        </plugin>

        <!-- Maven Failsafe Plugin (集成测试) -->
        <plugin>
            <groupId>org.apache.maven.plugins</groupId>
            <artifactId>maven-failsafe-plugin</artifactId>
            <version>3.0.0-M5</version>
            <configuration>
                <includes>
                    <include>**/*IntegrationTest.java</include>
                </includes>
            </configuration>
            <executions>
                <execution>
                    <goals>
                        <goal>integration-test</goal>
                        <goal>verify</goal>
                    </goals>
                </execution>
            </executions>
        </plugin>

        <!-- JaCoCo Plugin (代码覆盖率) -->
        <plugin>
            <groupId>org.jacoco</groupId>
            <artifactId>jacoco-maven-plugin</artifactId>
            <version>0.8.7</version>
            <executions>
                <execution>
                    <goals>
                        <goal>prepare-agent</goal>
                    </goals>
                </execution>
                <execution>
                    <id>report</id>
                    <phase>test</phase>
                    <goals>
                        <goal>report</goal>
                    </goals>
                </execution>
            </executions>
        </plugin>
    </plugins>
</build>
```

```bash
./gradlew build

```

### 测试相关命令

```bash
# 运行指定测试类
./gradlew test --tests UserServiceTest

# 运行指定测试方法
./gradlew test --tests UserServiceTest.testCreateUser

# 运行指定包下的测试
./gradlew test --tests com.example.service.*

# 运行匹配模式的测试
./gradlew test --tests "*IntegrationTest"

# 并行运行测试
./gradlew test --max-workers=4

# 生成测试报告
./gradlew test --rerun-tasks

./gradlew jacocoTestReport

./gradlew test --info
```

### 命令参数说明

| 参数 | 说明 |
|------|------|
| `--tests` | 指定测试类或方法 |
| `-x test` | 跳过测试 |
| `--max-workers` | 最大工作线程数 |
| `--rerun-tasks` | 重新运行任务 |
| `--info` | 显示详细信息 |
| `--debug` | 显示调试信息 |

### 常用配置

**build.gradle**

```groovy
plugins {
    id 'java'
    id 'jacoco'
}

test {
    useJUnitPlatform()
    
    // 并行执行
    maxParallelForks = 4
    
    // JVM参数
    jvmArgs '-Xmx1024m', '-Xms512m'
    
    // 测试日志
    testLogging {
        events "passed", "skipped", "failed"
        exceptionFormat "full"
    }
    
    // 包含和排除
    include '**/*Test.class'
    exclude '**/*IntegrationTest.class'
}

configurations {
    integrationTestImplementation.extendsFrom testImplementation
    integrationTestRuntimeOnly.extendsFrom testRuntimeOnly
}

sourceSets {
    integrationTest {
        compileClasspath += sourceSets.main.output
        runtimeClasspath += sourceSets.main.output
    }
}

task integrationTest(type: Test) {
    description = 'Runs integration tests.'
    group = 'verification'
    
    testClassesDirs = sourceSets.integrationTest.output.classesDirs
    classpath = sourceSets.integrationTest.runtimeClasspath
    
    shouldRunAfter test
    
    useJUnitPlatform()
    
    include '**/*IntegrationTest.class'
}

check.dependsOn integrationTest

// JaCoCo配置
jacoco {
    toolVersion = "0.8.7"
}

jacocoTestReport {
    dependsOn test
    
    reports {
        xml.enabled true
        html.enabled true
        csv.enabled false
    }
}

jacocoTestCoverageVerification {
    violationRules {
        rule {
            limit {
                minimum = 0.80
            }
        }
    }
}

check.dependsOn jacocoTestCoverageVerification
```

### 2. REST API集成测试

```java
@SpringBootTest
@AutoConfigureMockMvc
public class ApiIntegrationTest {
    
    @Autowired
    private MockMvc mockMvc;
    
    @Autowired
    private ObjectMapper objectMapper;
    
    @Test
    public void testCreateUser() throws Exception {
        CreateUserRequest request = new CreateUserRequest(
            "test@example.com",
            "password123",
            "张三"
        );
        
        mockMvc.perform(post("/api/users")
                .contentType(MediaType.APPLICATION_JSON)
                .content(objectMapper.writeValueAsString(request)))
            .andExpect(status().isCreated())
            .andExpect(jsonPath("$.id").exists())
            .andExpect(jsonPath("$.email").value("test@example.com"))
            .andExpect(jsonPath("$.name").value("张三"));

    public void testGetUser() throws Exception {
        mockMvc.perform(get("/api/users/1"))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.id").value(1))
            .andExpect(jsonPath("$.email").exists());
    }
    
    @Test
    public void testUpdateUser() throws Exception {
        UpdateUserRequest request = new UpdateUserRequest(
            "new@example.com",
            "李四"
        );
        
        mockMvc.perform(put("/api/users/1")
                .contentType(MediaType.APPLICATION_JSON)
                .content(objectMapper.writeValueAsString(request)))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.email").value("new@example.com"))
            .andExpect(jsonPath("$.name").value("李四"));
    }
    
    @Test
    public void testDeleteUser() throws Exception {
        mockMvc.perform(delete("/api/users/1"))
            .andExpect(status().isNoContent());
    }
}
```

```java
@SpringBootTest
public class AsyncIntegrationTest {
    
    @Autowired
    private AsyncService asyncService;
    
    @Test
    public void testAsyncMethod() throws InterruptedException, ExecutionException {
        CompletableFuture<String> future = asyncService.asyncMethod("test");
        
        String result = future.get(5, TimeUnit.SECONDS);
        assertEquals("test processed", result);
    }
    
    @Test
    public void testAsyncMethodWithException() {
        CompletableFuture<String> future = asyncService.asyncMethodWithError();
        
        assertThrows(ExecutionException.class, () -> {
            future.get(5, TimeUnit.SECONDS);
        });
    }
}

@Service
public class AsyncService {
    
    @Async
    public CompletableFuture<String> asyncMethod(String input) {
        try {
            Thread.sleep(1000);
            return CompletableFuture.completedFuture(input + " processed");
        } catch (InterruptedException e) {
            throw new RuntimeException(e);
        }
    }
    
    @Async
    public CompletableFuture<String> asyncMethodWithError() {
        return CompletableFuture.failedFuture(new RuntimeException("Async error"));
    }
}
```

### 8. 并发测试

```java
@SpringBootTest
@Testcontainers
public class ConcurrencyIntegrationTest {
    
    @Container
    static PostgreSQLContainer<?> postgres = new PostgreSQLContainer<>("postgres:14");
    
    @DynamicPropertySource
    static void postgresProperties(DynamicPropertyRegistry registry) {
        registry.add("spring.datasource.url", postgres::getJdbcUrl);
        registry.add("spring.datasource.username", postgres::getUsername);
        registry.add("spring.datasource.password", postgres::getPassword);
    }
    
    @Autowired
    private UserRepository userRepository;
    
    @Test
    public void testConcurrentUpdate() throws InterruptedException {
        User user = new User("test@example.com");
        user.setBalance(100);
        User savedUser = userRepository.save(user);
        
        int threadCount = 10;
        CountDownLatch latch = new CountDownLatch(threadCount);
        ExecutorService executor = Executors.newFixedThreadPool(threadCount);
        
        for (int i = 0; i < threadCount; i++) {
            executor.submit(() -> {
                try {
                    User userToUpdate = userRepository.findById(savedUser.getId()).orElseThrow();
                    userToUpdate.setBalance(userToUpdate.getBalance() + 10);
                    userRepository.save(userToUpdate);
                } finally {
                    latch.countDown();
                }
            });
        }
        
        latch.await(30, TimeUnit.SECONDS);
        executor.shutdown();
        
        User finalUser = userRepository.findById(savedUser.getId()).orElseThrow();
        assertEquals(200, finalUser.getBalance());
    }
}
```

## 排错指南

### 常见错误与解决方案

| 错误 | 原因 | 解决方案 |
|------|------|----------|
| **容器启动失败** | Docker未启动或端口被占用 | 检查Docker状态、释放端口 |
| **数据库连接失败** | 数据库未启动或配置错误 | 检查数据库状态、检查配置 |
| **测试超时** | 测试执行时间过长 | 增加超时时间、优化测试 |
| **内存不足** | JVM内存设置过小 | 调整JVM内存参数 |
| **测试数据冲突** | 测试之间共享数据 | 使用事务回滚、独立数据库 |
| **Mock失败** | Mock配置错误 | 检查Mock配置、使用正确的方法 |
| **端口冲突** | 多个测试使用相同端口 | 使用随机端口、释放端口 |
| **依赖注入失败** | Bean未正确配置 | 检查Bean配置、添加必要的注解 |
